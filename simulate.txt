// simulate.txt — Trading simulation logic
logToPanel("simulate.txt loaded");

let currentCycle = 1;
let cycleTimer = null;

function simulateCycle(cycle) {
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);

    // Pick a coin (replace with smarter logic later)
    const coin = pickRandomCoin(); 
    const buyPrice = getMockPrice(coin);
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Bought ${coin} at ${buyPrice} USDT (fee applied)`);

    // Mock profit/loss calculation
    const result = getMockResult(buyPrice);
    if (result >= 0) {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Result Profit (${result.toFixed(2)} USDT)`);
    } else {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Result Loss (${result.toFixed(2)} USDT)`);
    }

    // Update chart with balance + profit/loss
    updateChart(buyPrice, result);

    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) complete`);
}

// Start simulation with 1‑minute cycle interval
function startSimulation(totalCycles) {
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — initBot started for ${totalCycles} cycles (1‑minute interval)`);

    currentCycle = 1;
    cycleTimer = setInterval(() => {
        if (currentCycle > totalCycles) {
            clearInterval(cycleTimer);
            logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Simulation complete`);
            return;
        }

        simulateCycle(currentCycle);
        currentCycle++;
    }, 60000); // 60000 ms = 1 minute
}

// Wrapper so main.txt can call initBot()
function initBot(totalCycles = 10) {
    startSimulation(totalCycles);
}

// Helpers (mock logic for now)
function pickRandomCoin() {
    const coins = ["BTCUSDT", "ETHUSDT", "ADAUSDT", "UNIUSDT", "XLMUSDT", "PEPEUSDT"];
    return coins[Math.floor(Math.random() * coins.length)];
}

function getMockPrice(coin) {
    // Simple mock price generator
    return (Math.random() * 200).toFixed(2);
}

function getMockResult(price) {
    // Random profit/loss around ±10%
    const change = (Math.random() - 0.5) * 0.2;
    return price * change;
}

// Expose functions globally
window.startSimulation = startSimulation;
window.simulateCycle = simulateCycle;
window.initBot = initBot;
