// simulate.txt — Trading loop + Binance helpers + Chart initialization
logToPanel("simulate.txt loaded");

// Global balance history
window.balanceHistory = [];

// Initialize Chart.js once DOM is ready
document.addEventListener("DOMContentLoaded", function () {
    const canvas = document.getElementById("balanceChart");
    if (!canvas) {
        logWithTime("ERROR: balanceChart canvas not found");
        return;
    }
    const ctx = canvas.getContext("2d");
    window.balanceChart = new Chart(ctx, {
        type: "line",
        data: {
            labels: [],
            datasets: [{
                label: "Investment Balance",
                data: [],
                borderColor: "#f3ba2f",
                backgroundColor: "rgba(243,186,47,0.2)",
                fill: true,
                tension: 0.3
            }]
        },
        options: { responsive: true, plugins: { legend: { display: true } } }
    });
    logWithTime("Chart initialized successfully");
});

// Chart update helper (no parameter, uses balanceHistory)
function updateChart() {
    if (!window.balanceChart) {
        logWithTime("Chart not initialized yet");
        return;
    }
    window.balanceChart.data.labels.push(`Cycle ${window.balanceChart.data.labels.length + 1}`);
    window.balanceChart.data.datasets[0].data.push(window.balanceHistory[window.balanceHistory.length - 1]);
    window.balanceChart.update();
}
window.updateChart = updateChart;

// Logging helper with timestamp
function logWithTime(message) {
    const now = new Date();
    const timeStr = now.toLocaleTimeString("en-GB", { hour12: false });
    if (typeof window.logToPanel === "function") {
        window.logToPanel(`[INFO] ${timeStr} — ${message}`);
    } else {
        console.log(`[INFO] ${timeStr} — ${message}`);
    }
}

// … keep your seededRandom, evaluateCoin, getDynamicCoins, sleep, etc. unchanged …

// Main trading cycle
async function simulateCycle(cycleNum) {
    try {
        logWithTime(`simulateCycle(${cycleNum}) started`);

        const allCoins = await getDynamicCoins();
        const randIndex = Math.floor(seededRandom(cycleNum) * allCoins.length);
        const coin = allCoins[randIndex];
        const entryPrice = await evaluateCoin(coin, 1);

        // Deduct fee
        let investBalance = parseFloat(document.getElementById("investBalance").value);
        investBalance -= 0.10;
        document.getElementById("investBalance").value = investBalance.toFixed(2);

        logWithTime(`Cycle ${cycleNum}: Bought ${coin} at ${entryPrice.toFixed(2)} USDT (fee applied)`);

        await sleep(30000); // 30 sec lapse

        const currentPrice = await evaluateCoin(coin, 1);
        const profit = currentPrice - entryPrice;

        const profitElem = document.getElementById("liveProfitValue");
        profitElem.textContent = profit.toFixed(2) + " USDT";
        profitElem.style.color = profit >= 0 ? "#00FF00" : "#FF3333";

        logWithTime(`Cycle ${cycleNum}: Result ${profit >= 0 ? "Profit" : "Loss"} (${profit.toFixed(2)} USDT)`);

        window.balanceHistory.push(investBalance);
        updateChart();  // <-- no argument
        logWithTime(`simulateCycle(${cycleNum}) complete`);
    } catch (err) {
        logWithTime(`[ERROR] simulateCycle(${cycleNum}) crashed: ${err}`);
    }
}

// Expose globally
window.simulateCycle = simulateCycle;
