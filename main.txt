// main.txt — Lifecycle glue

function finalizeSummary() {
    const startBalance = parseFloat(document.getElementById("startBalance").value);
    const investBalance = parseFloat(document.getElementById("investBalance").value);
    const totalChange = investBalance - startBalance;
    const percentChange = (totalChange / startBalance) * 100;

    logToPanel(`Final Start Balance: ${startBalance.toFixed(2)} USDT`);
    logToPanel(`Final Investment Balance: ${investBalance.toFixed(2)} USDT`);
    logToPanel(`Net Change: ${totalChange.toFixed(2)} USDT (${percentChange.toFixed(2)}%)`);
}

async function initBot(cycles) {
    logToPanel("Bot started");
    for (let i = 1; i <= cycles; i++) {
        if (typeof simulateCycle === "function") {
            await simulateCycle(i);
        } else {
            logToPanel("ERROR: simulateCycle not found");
        }
    }
    finalizeSummary();
    logToPanel("Bot finished");
}

function stopBot() {
    logToPanel("Bot stopped");
}

// Expose globally
window.initBot = initBot;
window.stopBot = stopBot;
window.finalizeSummary = finalizeSummary;
