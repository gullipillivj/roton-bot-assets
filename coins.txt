import requests

BINANCE_TICKER = "https://api.binance.com/api/v3/ticker/price"

def get_btc_price():
    try:
        res = requests.get(BINANCE_TICKER, params={"symbol": "BTCUSDT"}, timeout=10)
        d = res.json()
        return float(d["price"])
    except Exception as e:
        print("[ERROR] Failed to fetch BTC price:", e)
        return None

if __name__ == "__main__":
    price = get_btc_price()
    if price:
        print(f"Current BTCUSDT price: {price}")
    else:
        print("DEBUG: No price returned from Binance")
