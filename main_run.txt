function safeParse(text) {
    try { return JSON.parse(text); } catch { return null; }
}

const USER_ADDR = "0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266";
const USER_PK   = "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80";

// === Refresh balances helper ===
async function refreshBalances(userAddress, label) {
    const resp = await fetch("/runbot/balances", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wallet_address: userAddress })
    });
    const parsed = safeParse(await resp.text());
    if (parsed?.contract) {
        const investBal = parsed.contract.fund_balance;

        // Initial stays fixed (whatever you set at page load)
        const initialVal = document.getElementById("startBalance").value || 0;
        document.getElementById("investBalance").value = investBal;

        logToPanel(`[${label}] Initial=${initialVal}, Investment=${investBal}`);
    }
}

// === Deposit ===
window.runDeposit = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        const resp = await fetch("/runbot/deposit", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress, private_key: USER_PK, amount })
        });
        const parsed = safeParse(await resp.text());
        if (parsed?.tx_hash)
            logToPanel(`[SUCCESS] Deposit confirmed | TxHash=${parsed.tx_hash}`);

        // ✅ Update textboxes
        const initialVal = document.getElementById("startBalance").value || 0;
        document.getElementById("investBalance").value = parsed.fund_balance;

        logToPanel(`[RESULT] Initial=${initialVal}, Investment=${parsed.fund_balance}`);

        // ✅ Refresh from contract
        await refreshBalances(userAddress, "POST-DEPOSIT");
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === DOM Ready ===
document.addEventListener("DOMContentLoaded", async () => {
    logToPanel("UI ready");

    // Set initial figure once (from db or hardcoded)
    document.getElementById("startBalance").value = 1000; // example figure
    document.getElementById("investBalance").value = 500; // example figure

    logToPanel(`[INIT] Initial=1000, Investment=500`);
