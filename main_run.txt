function logToPanel(msg) {
    const panel = document.getElementById("logPanel");
    panel.value += msg + "\n";
}

function safeParse(text) {
    try { return JSON.parse(text); } catch { return null; }
}

async function loadClientData(userId) {
    const resp = await fetch("/clients");
    const data = safeParse(await resp.text());
    const client = data?.clients?.find(c => c.userid === userId);
    if (!client) return logToPanel(`[WARN] No client found for ${userId}`);

    document.getElementById("startBalance").value = client.amount_deposited;
    document.getElementById("investBalance").value = client.current_balance;
    document.getElementById("profitTarget").value = client.profit_target;
    document.getElementById("stopLoss").value = client.stop_loss;

    logToPanel(`[INFO] Loaded client data for ${client.userid}`);
}

async function refreshBalances(userAddress) {
    const resp = await fetch("/runbot/balances", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wallet_address: userAddress })
    });
    const parsed = safeParse(await resp.text());
    if (parsed) {
        logToPanel(JSON.stringify(parsed));
        if (parsed.contract?.fund_balance !== undefined) {
            document.getElementById("investBalance").value = parsed.contract.fund_balance;
        }
    } else {
        logToPanel("[WARN] No data received from balances endpoint");
    }
}

async function runDeposit(userAddress, amount) {
    const resp = await fetch("/runbot/deposit", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wallet_address: userAddress, amount })
    });
    const parsed = safeParse(await resp.text());
    logToPanel(JSON.stringify(parsed));
    await refreshBalances(userAddress);
}

async function runWithdraw(userAddress, amount) {
    const resp = await fetch("/runbot/payout", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wallet_address: userAddress, amount })
    });
    const parsed = safeParse(await resp.text());
    logToPanel(JSON.stringify(parsed));
    await refreshBalances(userAddress);
}

async function changeUser(userId) {
    await loadClientData(userId);

    const userSelect = document.getElementById("userSelect");
    if (userSelect) {
        for (let opt of userSelect.options) {
            opt.style.backgroundColor = "";
            opt.style.color = "";
        }
        userSelect.options[userSelect.selectedIndex].style.backgroundColor = "lightgreen";
        userSelect.options[userSelect.selectedIndex].style.color = "black";
    }

    const resp = await fetch("/clients");
    const data = safeParse(await resp.text());
    const client = data?.clients?.find(c => c.userid === userId);
    if (!client) return;

    const USER_ADDR = client.wallet_address;
    document.getElementById("currentUserLabel").innerText = userId;
    logToPanel(`[INFO] Active user: ${userId}, Wallet=${USER_ADDR}`);
    await refreshBalances(USER_ADDR);
}

document.addEventListener("DOMContentLoaded", async () => {
    logToPanel("UI ready");
    await loadClientData("mockuser1");
    await refreshBalances("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266");

    document.getElementById("depositBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
        runDeposit("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
    });

    document.getElementById("withdrawBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
        runWithdraw("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
    });

    document.getElementById("changeUserBtn")?.addEventListener("click", async () => {
        const selectedUser = document.getElementById("userSelect").value;
        await changeUser(selectedUser);
    });
});
