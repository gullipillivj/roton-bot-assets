logToPanel("main_run.txt loaded");

function safeParse(text) {
    try { return JSON.parse(text); } catch { return null; }
}

const USER_ADDR = "0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266";
const USER_PK   = "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80";

// === Deposit ===
window.runDeposit = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        // Step 1: Approve
        const approveResp = await fetch("/runbot/approve", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress, private_key: USER_PK, amount })
        });
        const approveParsed = safeParse(await approveResp.text());
        if (approveParsed?.tx_hash) {
            logToPanel(`[SUCCESS] Approve confirmed | TxHash=${approveParsed.tx_hash}`);
        }

        // Step 2: Deposit
        const depositResp = await fetch("/runbot/deposit", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress, private_key: USER_PK, amount })
        });
        const depositParsed = safeParse(await depositResp.text());
        if (depositParsed?.tx_hash) {
            logToPanel(`[SUCCESS] Deposit confirmed | TxHash=${depositParsed.tx_hash}`);
        }

        // Step 3: Update textbox values (add amount)
        const currentStart = parseInt(document.getElementById("startBalance").value) || 0;
        document.getElementById("startBalance").value = currentStart + amount;

        const currentInvest = parseInt(document.getElementById("investBalance").value) || 0;
        document.getElementById("investBalance").value = currentInvest + amount;

        // Step 4: Sync DB
        await fetch("/runbot/update_client", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                userid: "mockuser1",
                amount_deposited: document.getElementById("startBalance").value,
                current_balance: document.getElementById("investBalance").value
            })
        });
        logToPanel("[SUCCESS] Client balances updated in DB after deposit");

        // Step 5: Refresh balances from contract
        const balResp = await fetch("/runbot/balances", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress })
        });
        const balData = safeParse(await balResp.text());
        if (balData?.contract) {
            document.getElementById("investBalance").value = balData.contract.fund_token_balance;
            document.getElementById("startBalance").value = balData.contract.eth_balance_wei;
            logToPanel(`[BALANCES AFTER DEPOSIT] ${JSON.stringify(balData.contract)}`);
        }
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === Withdraw ===
window.runWithdraw = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting withdraw for ${userAddress}, amount=${amount}`);

        // Step 1: Payout
        const payoutResp = await fetch("/runbot/payout", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress, private_key: USER_PK, amount })
        });
        const payoutParsed = safeParse(await payoutResp.text());
        if (payoutParsed?.tx_hash) {
            logToPanel(`[SUCCESS] Withdraw confirmed | TxHash=${payoutParsed.tx_hash}`);
        }

        // Step 2: Update textbox values (subtract amount)
        const currentStart = parseInt(document.getElementById("startBalance").value) || 0;
        document.getElementById("startBalance").value = Math.max(currentStart - amount, 0);

        const currentInvest = parseInt(document.getElementById("investBalance").value) || 0;
        document.getElementById("investBalance").value = Math.max(currentInvest - amount, 0);

        // Step 3: Sync DB
        await fetch("/runbot/update_client", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                userid: "mockuser1",
                amount_deposited: document.getElementById("startBalance").value,
                current_balance: document.getElementById("investBalance").value
            })
        });
        logToPanel("[SUCCESS] Client balances updated in DB after withdraw");

        // Step 4: Refresh balances from contract
        const balResp = await fetch("/runbot/balances", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress })
        });
        const balData = safeParse(await balResp.text());
        if (balData?.contract) {
            document.getElementById("investBalance").value = balData.contract.fund_token_balance;
            document.getElementById("startBalance").value = balData.contract.eth_balance_wei;
            logToPanel(`[BALANCES AFTER WITHDRAW] ${JSON.stringify(balData.contract)}`);
        }
    } catch (e) {
        logToPanel(`[FAILURE] Withdraw ${amount} | Error=${e.message}`);
    }
};

// === DOM Ready ===
document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");

    document.getElementById("depositBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
        window.runDeposit(USER_ADDR, amt);
    });

    document.getElementById("withdrawBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
        window.runWithdraw(USER_ADDR, amt);
    });
});
