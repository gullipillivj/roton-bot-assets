// main_run.txt — central loader
logToPanel("main_run.txt loaded");

// Load child modules in order so globals are ready
(async () => {
    await import("./logger.txt");
    await import("./contracts.txt");
    await import("./coins.txt");
    await import("./config.txt");
    await import("./trade_logic.txt");
    await import("./db_access.txt");
    await import("./chart.txt");
    await import("./simulate_run.txt");
})();

document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");

    // Start Trading button
    const startBtn = document.getElementById("startBtn");
    if (startBtn) {
        startBtn.addEventListener("click", () => {
            if (typeof window.initBot_run === "function") {
                window.initBot_run(10); // always run 10 cycles
            } else {
                logToPanel("ERROR: initBot not found");
            }
        });
    }

    // Stop Trading button
    const stopBtn = document.getElementById("stopBtn");
    if (stopBtn) {
        stopBtn.addEventListener("click", () => {
            if (typeof window.stopBot_run === "function") {
                window.stopBot_run();
            } else {
                logToPanel("ERROR: stopBot not found");
            }
        });
    }
});
