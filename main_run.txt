// === Helper: safe JSON parse ===
function safeParse(text) {
    try {
        return JSON.parse(text);
    } catch {
        return null;
    }
}

// === Constants (wallet + private key) ===
const USER_ADDR = "0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266";
const USER_PK   = "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80";

// === Deposit ===
window.runDeposit = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        const response = await fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/deposit", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                wallet_address: userAddress,
                private_key: USER_PK,
                amount: amount
            })
        });

        const text = await response.text();
        const parsed = safeParse(text);
        if (parsed && parsed.tx_hash) {
            logToPanel(`[SUCCESS] Deposit confirmed | TxHash=${parsed.tx_hash}`);
        } else {
            logToPanel(`[RESULT] Deposit response=${text}`);
        }

        // Refresh balances
        const balResponse = await fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/balances", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress })
        });
        const balText = await balResponse.text();
        const balData = safeParse(balText);

        if (balData) {
            document.getElementById("investBalance").value = balData.fund_token_balance;
            document.getElementById("startBalance").value = balData.eth_balance_wei;
            logToPanel(`[BALANCES AFTER DEPOSIT] ${JSON.stringify(balData)}`);
        } else {
            logToPanel(`[BALANCES AFTER DEPOSIT] ${balText}`);
        }

        // Persist into clients.db (optional, backend must support /update_client)
        try {
            await fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/update_client", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    userid: "mockuser1",
                    amount_deposited: balData ? balData.eth_balance_wei : null,
                    current_balance: balData ? balData.fund_token_balance : null
                })
            });
            logToPanel("[SUCCESS] Client balances updated in DB");
        } catch (dbErr) {
            logToPanel(`[WARN] Could not update clients.db: ${dbErr.message}`);
        }
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === Bot start/stop handlers ===
let botInterval = null;

window.initBot = function(intervalSeconds = 10) {
    if (botInterval) {
        clearInterval(botInterval);
    }
    logToPanel(`[BOT] Trading loop started, interval=${intervalSeconds}s`);

    botInterval = setInterval(async () => {
        try {
            // Example trading cycle: approve → swap → payout → balances
            await window.runApprove(USER_ADDR, 1000, USER_PK);
            await window.runSwap(USER_ADDR, 200, USER_PK);
            await window.runPayout(USER_ADDR, 500, USER_PK);

            const balResponse = await fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/balances", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ wallet_address: USER_ADDR })
            });
            const balText = await balResponse.text();
            const balData = safeParse(balText);

            if (balData) {
                document.getElementById("investBalance").value = balData.fund_token_balance;
                document.getElementById("startBalance").value = balData.eth_balance_wei;
                logToPanel(`[BOT BALANCES] ${JSON.stringify(balData)}`);
            } else {
                logToPanel(`[BOT BALANCES] ${balText}`);
            }
        } catch (err) {
            logToPanel(`[BOT ERROR] ${err.message}`);
        }
    }, intervalSeconds * 1000);
};

window.stopBot = function() {
    if (botInterval) {
        clearInterval(botInterval);
        botInterval = null;
        logToPanel("[BOT] Trading loop stopped");
    }
};
