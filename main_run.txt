// ---- Hardcoded constants for mockuser1 ----
const USER_ID  = "mockuser1";
const USER_ADDR = "0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266";
const USER_PK   = "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80";
const BASE_URL  = "https://almanac-sierra-rundown.ngrok-free.dev";

function logToPanel(msg) {
    const panel = document.getElementById("logPanel");
    panel.value += msg + "\n";
}

function safeParse(text) {
    try { return JSON.parse(text); } catch { return null; }
}

async function loadClientData(userId) {
    const resp = await fetch(`${BASE_URL}/clients`);
    const data = safeParse(await resp.text());
    const client = data?.clients?.find(c => c.userid === userId);
    if (!client) return logToPanel(`[WARN] No client found for ${userId}`);

    document.getElementById("startBalance").value = client.amount_deposited;
    document.getElementById("investBalance").value = client.current_balance;
    document.getElementById("profitTarget").value = client.profit_target;
    document.getElementById("stopLoss").value = client.stop_loss;

    logToPanel(`[INFO] Loaded client data for ${client.userid}`);
}

async function refreshBalances() {
    const resp = await fetch(`${BASE_URL}/runbot/deposit`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wallet_address: USER_ADDR, private_key: USER_PK, amount: 0 })
    });
    const parsed = safeParse(await resp.text());
    if (parsed?.fund_token_balance !== undefined) {
        document.getElementById("startBalance").value = parsed.fund_token_balance;
        document.getElementById("investBalance").value = parsed.fund_token_balance;
        logToPanel(`[INFO] Fund balance: ${parsed.fund_token_balance}, ETH=${parsed.eth_balance_wei}`);
    } else {
        logToPanel("[WARN] No balance data received");
    }
}

async function runDeposit(amount) {
    try {
        // Step 1: Approve
        const approveResp = await fetch(`${BASE_URL}/runbot/approve`, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: USER_ADDR, private_key: USER_PK, amount })
        });
        const approveParsed = safeParse(await approveResp.text());
        if (approveParsed?.tx_hash) {
            logToPanel(`[INFO] Approve executed, TX=${approveParsed.tx_hash}`);
        }

        // Step 2: Deposit
        const depositResp = await fetch(`${BASE_URL}/runbot/deposit`, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: USER_ADDR, private_key: USER_PK, amount })
        });
        const depositParsed = safeParse(await depositResp.text());
        if (depositParsed) {
            logToPanel(`[INFO] Deposit executed, TX=${depositParsed.tx_hash}, Fund=${depositParsed.fund_token_balance}, ETH=${depositParsed.eth_balance_wei}`);
            document.getElementById("startBalance").value = depositParsed.fund_token_balance;
            document.getElementById("investBalance").value = depositParsed.fund_token_balance;
        }
    } catch (err) {
        logToPanel(`[ERROR] Deposit failed: ${err.message}`);
    }
}

async function runWithdraw(amount) {
    try {
        const resp = await fetch(`${BASE_URL}/runbot/payout`, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: USER_ADDR, private_key: USER_PK, amount })
        });
        const parsed = safeParse(await resp.text());
        if (parsed) {
            logToPanel(`[INFO] Payout executed, TX=${parsed.tx_hash}`);
        }
        await refreshBalances();
    } catch (err) {
        logToPanel(`[ERROR] Withdraw failed: ${err.message}`);
    }
}

document.addEventListener("DOMContentLoaded", async () => {
    logToPanel("UI ready");
    await loadClientData(USER_ID);
    await refreshBalances();

    document.getElementById("depositBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
        runDeposit(amt);
    });

    document.getElementById("withdrawBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
        runWithdraw(amt);
    });

    // ✅ Start Trading button
    document.getElementById("startTradingBtn")?.addEventListener("click", async () => {
        const startBal = parseFloat(document.getElementById("startBalance").value) || 0;
        const investBal = parseFloat(document.getElementById("investBalance").value) || 0;

        if (investBal > startBal) {
            logToPanel("[WARN] Investment balance must be less than or equal to starting balance to start trading");
            return;
        }

        logToPanel("[INFO] Start Trading button pressed — launching simulation");
        setStatus("RUNNING");

        // Kick off the simulate_run loop (10 cycles, with thresholds + status updates)
        window.startSimulation(10);
    });
});
