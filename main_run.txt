// === Balance check helper (raw text) ===
window.checkBalances = async function(userAddress) {
    try {
        const response = await fetch("/balances");
        const text = await response.text();
        logToPanel(`[BALANCES] ${text}`);
    } catch (e) {
        logToPanel("[ERROR] checkBalances failed: " + e.message);
    }
};

// === Deposit with backend call + UI + DB update ===
window.runDeposit = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        const response = await fetch("/runbot/deposit", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                wallet_address: userAddress,
                private_key: "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80",
                amount: amount
            })
        });

        const text = await response.text();
        let parsed;
        try {
            parsed = JSON.parse(text);
            logToPanel(`[SUCCESS] Deposit confirmed | TxHash=${parsed.tx_hash}`);
        } catch (err) {
            logToPanel(`[RESULT] Deposit response=${text}`);
        }

        // Refresh balances
        const balResponse = await fetch("/runbot/balances");
        const balData = await balResponse.json();

        // Update UI textboxes
        document.getElementById("investBalance").value = balData.fund_token_balance;
        document.getElementById("startBalance").value = balData.eth_balance_wei;

        // Log balances
        logToPanel(`[BALANCES AFTER DEPOSIT] ${JSON.stringify(balData)}`);

        // Persist into clients.db (stubbed until backend route exists)
        try {
            await fetch("/runbot/update_client", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    userid: "mockuser1",
                    amount_deposited: balData.eth_balance_wei,
                    current_balance: balData.fund_token_balance
                })
            });
            logToPanel("[SUCCESS] Client balances updated in DB");
        } catch (dbErr) {
            logToPanel(`[WARN] Could not update clients.db: ${dbErr.message}`);
        }
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === Load client data silently ===
async function loadClientData(username = "mockuser1") {
    try {
        const response = await fetch("/clients");
        const data = await response.json();
        const client = data.clients.find(c => c.userid === username || c.name === username);
        if (!client) return;

        document.getElementById("startBalance").value = client.amount_deposited || 0;
        document.getElementById("investBalance").value = client.current_balance || 0;
        document.getElementById("profitTarget").value = client.profit_target || 1;
        document.getElementById("stopLoss").value = client.stop_loss || 0.5;
        document.getElementById("currentUserLabel").textContent = client.name;
    } catch (e) {
        logToPanel("[ERROR] loadClientData failed: " + e.message);
    }
}

window.loadClientData = loadClientData;

document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");
    window.loadClientData("mockuser1");

    // Bot controls
    const startBtn = document.getElementById("startBtn");
    if (startBtn) {
        startBtn.addEventListener("click", () => {
            if (typeof window.initBot === "function") {
                window.initBot(10);
                logToPanel("[INFO] Bot started");
            } else {
                logToPanel("[ERROR] initBot function not found");
            }
        });
    }

    const stopBtn = document.getElementById("stopBtn");
    if (stopBtn) {
        stopBtn.addEventListener("click", () => {
            if (typeof window.stopBot === "function") {
                window.stopBot();
                logToPanel("[INFO] Bot stopped");
            } else {
                logToPanel("[ERROR] stopBot function not found");
            }
        });
    }

    // Account actions — direct Anvil address
    const depositBtn = document.getElementById("depositBtn");
    if (depositBtn) {
        depositBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
            window.runDeposit("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
        });
    }

    const withdrawBtn = document.getElementById("withdrawBtn");
    if (withdrawBtn) {
        withdrawBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
            window.runWithdraw("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
        });
    }
});
