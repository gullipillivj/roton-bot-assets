// main_run.txt — central loader
logToPanel("main_run.txt loaded");

// Wire UI after DOM is ready
document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");

    const startBtn = document.getElementById("startBtn");
    if (startBtn) {
        startBtn.addEventListener("click", () => {
            if (typeof window.initBot_run === "function") {
                // Use the active user set in db_access.txt
                const user = window.activeUserId || "0xYourUserWallet";
                window.initBot_run(10, user);
            } else {
                logToPanel("ERROR: initBot_run not found");
            }
        });
    }

    const stopBtn = document.getElementById("stopBtn");
    if (stopBtn) {
        stopBtn.addEventListener("click", () => {
            if (typeof window.stopBot_run === "function") {
                const user = window.activeUserId || "0xYourUserWallet";
                window.stopBot_run(user);
            } else {
                logToPanel("ERROR: stopBot_run not found");
            }
        });
    }
});
