// === Helper: safe JSON parse ===
function safeParse(text) {
    try {
        return JSON.parse(text);
    } catch {
        return null;
    }
}

// === Constants (wallet + private key) ===
const USER_ADDR = "0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266";
const USER_PK   = "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80";

// === Deposit ===
window.runDeposit = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        // Step 1: Call deposit route
        const response = await fetch("/runbot/deposit", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                wallet_address: userAddress,
                private_key: USER_PK,
                amount: amount
            })
        });

        const parsed = safeParse(await response.text());
        if (parsed && parsed.tx_hash) {
            logToPanel(`[SUCCESS] Deposit confirmed | TxHash=${parsed.tx_hash}`);
        } else {
            logToPanel(`[RESULT] Deposit response=${JSON.stringify(parsed)}`);
        }

        // Step 2: Refresh balances
        const balResponse = await fetch("/runbot/balances", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress })
        });
        const balData = safeParse(await balResponse.text());

        if (balData && balData.contract) {
            document.getElementById("investBalance").value = balData.contract.fund_token_balance;
            document.getElementById("startBalance").value = balData.contract.eth_balance_wei;

            logToPanel(`[BALANCES AFTER DEPOSIT] ${JSON.stringify(balData)}`);

            // Step 3: Sync DB
            await fetch("/runbot/update_client", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    userid: "mockuser1",
                    amount_deposited: balData.contract.eth_balance_wei,
                    current_balance: balData.contract.fund_token_balance
                })
            });
            logToPanel("[SUCCESS] Client balances updated in DB after deposit");
        }
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === DOM Ready ===
document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");

    const depositBtn = document.getElementById("depositBtn");
    if (depositBtn) {
        depositBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
            window.runDeposit(USER_ADDR, amt);
        });
    }

    const withdrawBtn = document.getElementById("withdrawBtn");
