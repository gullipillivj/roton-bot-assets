// === Balance check helper (raw text) ===
window.checkBalances = async function(userAddress) {
    try {
        const response = await fetch("/balances");
        const text = await response.text();   // no JSON parsing
        logToPanel(`[BALANCES] ${text}`);
    } catch (e) {
        logToPanel("[ERROR] checkBalances failed: " + e.message);
    }
};

// === Deposit with receipt + polling ===
window.runDeposit = async function(userAddress, amount) {
    try {
        const amtWei = web3.utils.toWei(amount.toString());
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        await approveToken(userTokenAddress, amtWei, userAddress);
        await depositToFund(userTokenAddress, amtWei, userAddress)
            .on('receipt', receipt => {
                logToPanel(`[RESULT] Deposit TxHash=${receipt.transactionHash || "N/A"}`);
            });

        // Poll balances until updated
        let initialText = "";
        let pollCount = 0;
        const pollInterval = setInterval(async () => {
            pollCount++;
            try {
                const response = await fetch("/balances");
                const text = await response.text();

                if (initialText === "") {
                    initialText = text;
                    logToPanel(`[DEBUG] Initial balances=${initialText}`);
                } else if (text !== initialText) {
                    clearInterval(pollInterval);
                    logToPanel(`[SUCCESS] Amount deposited ${amount} | Updated balances=${text}`);
                } else if (pollCount > 10) {
                    clearInterval(pollInterval);
                    logToPanel(`[WARN] Deposit not reflected after polling. Last balances=${text}`);
                }
            } catch (e) {
                logToPanel(`[ERROR] Balance poll failed: ${e.message}`);
                clearInterval(pollInterval);
            }
        }, 5000);

    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");
    window.loadClientData("mockuser1");

    // Bot controls
    const startBtn = document.getElementById("startBtn");
    if (startBtn) {
        startBtn.addEventListener("click", () => {
            if (typeof window.initBot === "function") {
                window.initBot(10);
                logToPanel("[INFO] Bot started");
            } else {
                logToPanel("[ERROR] initBot function not found");
            }
        });
    }

    const stopBtn = document.getElementById("stopBtn");
    if (stopBtn) {
        stopBtn.addEventListener("click", () => {
            if (typeof window.stopBot === "function") {
                window.stopBot();
                logToPanel("[INFO] Bot stopped");
            } else {
                logToPanel("[ERROR] stopBot function not found");
            }
        });
    }

    // Account actions — direct Anvil address
    const depositBtn = document.getElementById("depositBtn");
    if (depositBtn) {
        depositBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
            window.runDeposit("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
        });
    }

    const withdrawBtn = document.getElementById("withdrawBtn");
    if (withdrawBtn) {
        withdrawBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
            window.runWithdraw("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
        });
    }

    // Global balance polling every 10 seconds
    setInterval(() => {
        window.checkBalances("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266");
    }, 10000);
});
