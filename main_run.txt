// === Balance check helper ===
window.checkBalances = async function(userAddress) {
    try {
        const response = await fetch("/balances");
        const data = await response.json();
        logToPanel(`[BALANCES] Donor UserToken=${data.donor_balance_user_token}, Donor TargetToken=${data.donor_balance_target_token}, Fund UserToken=${data.fund_balance_user_token}, Fund TargetToken=${data.fund_balance_target_token}, Charity UserToken=${data.charity_balance_user_token}, Charity TargetToken=${data.charity_balance_target_token}`);
    } catch (e) {
        logToPanel("[ERROR] checkBalances failed: " + e.message);
    }
};

// === Deposit function using helpers with receipt logging ===
window.runDeposit = async function(userAddress, amount) {
    try {
        const amtWei = web3.utils.toWei(amount.toString());
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        await approveToken(userTokenAddress, amtWei, userAddress);
        await depositToFund(userTokenAddress, amtWei, userAddress)
            .on('receipt', receipt => {
                const txHash = receipt.transactionHash || "N/A";
                const block  = receipt.blockNumber || "N/A";
                const gas    = receipt.gasUsed || "N/A";
                const status = receipt.status ? "SUCCESS" : "FAILED";
                logToPanel(`[RESULT] Deposit — TxHash: ${txHash}, Block: ${block}, Gas: ${gas}, Status: ${status}`);
            });

        logToPanel(`[SUCCESS] Deposit ${amount} → Fund`);

        await window.checkBalances(userAddress);
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === Withdraw function using helpers with receipt logging ===
window.runWithdraw = async function(userAddress, amount) {
    try {
        const amtWei = web3.utils.toWei(amount.toString());
        logToPanel(`[INFO] Starting withdraw for ${userAddress}, amount=${amount}`);

        await withdrawFromFund(userTokenAddress, amtWei, userAddress)
            .on('receipt', receipt => {
                const txHash = receipt.transactionHash || "N/A";
                const block  = receipt.blockNumber || "N/A";
                const gas    = receipt.gasUsed || "N/A";
                const status = receipt.status ? "SUCCESS" : "FAILED";
                logToPanel(`[RESULT] Withdraw — TxHash: ${txHash}, Block: ${block}, Gas: ${gas}, Status: ${status}`);
            });

        logToPanel(`[SUCCESS] Withdraw ${amount} | User=${userAddress}`);

        await window.checkBalances(userAddress);
    } catch (e) {
        logToPanel(`[FAILURE] Withdraw ${amount} | Error=${e.message}`);
    }
};

// === Load client data function ===
async function loadClientData(username = "mockuser1") {
    try {
        const response = await fetch("/clients");
        const data = await response.json();
        const client = data.clients.find(c => c.userid === username || c.name === username);
        if (!client) {
            logToPanel(`[WARN] ${username} not found in DB`);
            return;
        }

        document.getElementById("startBalance").value = client.amount_deposited || 0;
        document.getElementById("investBalance").value = client.current_balance || 0;
        document.getElementById("profitTarget").value = client.profit_target || 1;
        document.getElementById("stopLoss").value = client.stop_loss || 0.5;
        document.getElementById("currentUserLabel").textContent = client.name;
    } catch (e) {
        logToPanel("[ERROR] loadClientData failed: " + e.message);
    }
}

// Expose globally
window.loadClientData = loadClientData;

document.addEventListener("DOMContentLoaded", () => {
    logToPanel("UI ready");
    window.loadClientData("mockuser1");

    // Bot controls
    const startBtn = document.getElementById("startBtn");
    if (startBtn) {
        startBtn.addEventListener("click", () => {
            if (typeof window.initBot === "function") {
                window.initBot(10);
                logToPanel("[INFO] Bot started");
            } else {
                logToPanel("[ERROR] initBot function not found");
            }
        });
    }

    const stopBtn = document.getElementById("stopBtn");
    if (stopBtn) {
        stopBtn.addEventListener("click", () => {
            if (typeof window.stopBot === "function") {
                window.stopBot();
                logToPanel("[INFO] Bot stopped");
            } else {
                logToPanel("[ERROR] stopBot function not found");
            }
        });
    }

    // Account actions — direct Anvil address
    const depositBtn = document.getElementById("depositBtn");
    if (depositBtn) {
        depositBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
            window.runDeposit("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
        });
    }

    const withdrawBtn = document.getElementById("withdrawBtn");
    if (withdrawBtn) {
        withdrawBtn.addEventListener("click", () => {
            const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
            window.runWithdraw("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266", amt);
        });
    }

    // Start balance polling every 10 seconds
    setInterval(() => {
        window.checkBalances("0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266");
    }, 10000);
});
