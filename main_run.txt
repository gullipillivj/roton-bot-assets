// main_run.txt — UI + trading glue
logToPanel("main_run.txt loaded");

function safeParse(text) {
    try { return JSON.parse(text); } catch { return null; }
}

const USER_ADDR = "0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266";
const USER_PK   = "0xac0974bec39a17e36ba4a6b4d238ff944bacb478cbed5efcae784d7bf4f2ff80";

// === Refresh balances helper ===
async function refreshBalances(userAddress, label) {
    const balResp = await fetch("/runbot/balances", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wallet_address: userAddress })
    });
    const balData = safeParse(await balResp.text());
    if (balData?.contract) {
        document.getElementById("investBalance").value = balData.contract.fund_token_balance;
        document.getElementById("startBalance").value = balData.contract.eth_balance_wei;
        logToPanel(`[${label}] Balances: ${JSON.stringify(balData.contract)}`);
    }
}

// === Deposit ===
window.runDeposit = async function(userAddress, amount) {
    try {
        logToPanel(`[INFO] Starting deposit for ${userAddress}, amount=${amount}`);

        // Approve
        const approveResp = await fetch("/runbot/approve", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress, private_key: USER_PK, amount })
        });
        const approveParsed = safeParse(await approveResp.text());
        if (approveParsed?.tx_hash) logToPanel(`[SUCCESS] Approve confirmed | TxHash=${approveParsed.tx_hash}`);

        // Deposit
        const depositResp = await fetch("/runbot/deposit", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ wallet_address: userAddress, private_key: USER_PK, amount })
        });
        const depositParsed = safeParse(await depositResp.text());
        if (depositParsed?.tx_hash) logToPanel(`[SUCCESS] Deposit confirmed | TxHash=${depositParsed.tx_hash}`);

        // ✅ Update textboxes directly from response
        if (depositParsed?.fund_token_balance !== undefined) {
            document.getElementById("investBalance").value = depositParsed.fund_token_balance;
        }
        if (depositParsed?.eth_balance_wei !== undefined) {
            document.getElementById("startBalance").value = depositParsed.eth_balance_wei;
        }

        // Sync DB
        await fetch("/runbot/update_client", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                userid: "mockuser1",
                amount_deposited: document.getElementById("startBalance").value,
                current_balance: document.getElementById("investBalance").value
            })
        });
        logToPanel("[SUCCESS] Client balances updated in DB after deposit");

        // ✅ Refresh from contract
        await refreshBalances(userAddress, "POST-DEPOSIT");
    } catch (e) {
        logToPanel(`[FAILURE] Deposit ${amount} | Error=${e.message}`);
    }
};

// === DOM Ready ===
document.addEventListener("DOMContentLoaded", async () => {
    logToPanel("UI ready");

    // ✅ Initial balances
    await refreshBalances(USER_ADDR, "INIT");

    document.getElementById("depositBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("depositAmount").value) || 0;
        window.runDeposit(USER_ADDR, amt);
    });

    document.getElementById("withdrawBtn")?.addEventListener("click", () => {
        const amt = parseFloat(document.getElementById("withdrawAmount").value) || 0;
