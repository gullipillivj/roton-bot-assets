logToPanel("pancake_tx.txt loaded")

# === Config ===
COMMISSION = 5
ROUTER_ADDRESS = "0x9ac64cc6e4415144c455bd8e4837fea55603e5c3"
CONFIRM_DELAY = 7000
SLIPPAGE = 1

# === Confirm helper ===
async def confirmTransaction(beforeAmount, expectedDelta, type, currentUser):
    startBalanceBox = document.getElementById("startBalance")

    setTimeout(async () => {
        afterBal = await window.fund.methods.balanceOf(currentUser).call()
        afterAmount = parseFloat(afterBal) || 0
        startBalanceBox.value = afterAmount

        diff = type == "Deposit" ? afterAmount - beforeAmount : beforeAmount - afterAmount

        if (Math.abs(diff - expectedDelta) < 0.0001):
            logToPanel("[RESULT] " + type + ": SUCCESS (Δ=" + diff + ")")
        else:
            logToPanel("[RESULT] " + type + ": FAILED (Δ=" + diff + ", expected " + expectedDelta + ")")
    }, CONFIRM_DELAY)

# === Deposit ===
async def runDeposit(username, amount):
    try:
        currentUser = userAddressMap[username]
        netAmount = amount - COMMISSION

        if (amount <= COMMISSION):
            logToPanel("[WARN] Deposit amount must be greater than commission fee")
            return

        beforeBal = await window.fund.methods.balanceOf(currentUser).call()
        beforeAmount = parseFloat(beforeBal) || 0
        document.getElementById("startBalance").value = beforeAmount

        payload = {
            "token_out": window.userTokenAddress,
            "amount_bnb": netAmount,
            "slippage": SLIPPAGE,
            "router_address": ROUTER_ADDRESS
        }

        await fetch("/api/deposit", {
            "method": "POST",
            "headers": { "Content-Type": "application/json" },
            "body": JSON.stringify(payload)
        })

        confirmTransaction(beforeAmount, netAmount, "Deposit", currentUser)
    except err:
        logToPanel("[ERROR] Deposit failed: " + err.message)

# === Withdraw ===
async def runWithdraw(username, amount):
    try:
        currentUser = userAddressMap[username]

        if (amount <= 0):
            logToPanel("[WARN] Withdraw amount must be greater than 0")
            return

        beforeBal = await window.fund.methods.balanceOf(currentUser).call()
        beforeAmount = parseFloat(beforeBal) || 0
        document.getElementById("startBalance").value = beforeAmount

        payload = {
            "token_in": window.userTokenAddress,
            "amount_in": amount,
            "slippage": SLIPPAGE,
            "router_address": ROUTER_ADDRESS
        }

        await fetch("/api/withdraw", {
            "method": "POST",
            "headers": { "Content-Type": "application/json" },
            "body": JSON.stringify(payload)
        })

        confirmTransaction(beforeAmount, amount, "Withdraw", currentUser)
    except err:
        logToPanel("[ERROR] Withdraw failed: " + err.message)

# === Expose globally (coins.txt style) ===
window.runDeposit = runDeposit
window.runWithdraw = runWithdraw
