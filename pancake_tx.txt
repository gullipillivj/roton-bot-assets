// === Helpers ===
function setStatus(state) {
  const label = document.getElementById("statusLabel");
  if (label) label.textContent = `STATE: ${state}`;
}

function logResult(action, data) {
  const txHash = data.tx_hash || "N/A";
  const status = data.status || "UNKNOWN";
  logToPanel(`[RESULT] ${action} — TxHash: ${txHash}, Status: ${status}`);
}

// === Backend calls (Flask handles Anvil) ===
async function approveToken(amount, userAddress, privateKey) {
  try {
    const res = await fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/approve", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ wallet_address: userAddress, private_key: privateKey, amount })
    });
    const data = await res.json();
    logResult("Approve", data);
    return data;
  } catch (err) {
    logToPanel("[ERROR] Approve failed: " + err.message);
    throw err;
  }
}

// === Wrappers for bot loop ===
async function runApprove(userAddress, amount, privateKey) {
  return approveToken(amount, userAddress, privateKey);
}

async function runSwap(userAddress, amount, privateKey) {
  return swapToTarget(amount, userAddress, privateKey);
}

async function runPayout(userAddress, amount, privateKey) {
  return swapToUser(amount, userAddress, privateKey);
}

// === Wrappers for UI buttons ===
async function runDeposit(userAddress, amount, privateKey) {
  setStatus("DEPOSITING");
  await runApprove(userAddress, amount, privateKey);
  await depositToFund(amount, userAddress, privateKey);
  setStatus("COMPLETE");
}

async function runWithdraw(userAddress, amount, privateKey) {
  setStatus("WITHDRAWING");
  await withdrawFromFund(amount, userAddress, privateKey);
  setStatus("COMPLETE");
}
