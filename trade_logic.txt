async function runCycle(userid) {
    const client = await getClientInfo(userid);
    if (!client || client.current_balance <= 0) {
        logWithTime(`[Cycle] No funds available for user ${userid}`);
        return;
    }

    logWithTime(`[Config] ProfitTarget=${client.profit_target}% StopLoss=${client.stop_loss}%`);

    // Example trade using contracts + tokens
    logWithTime(`[Trade] Executing swap for ${client.userid}`);
    logWithTime(`[Router] Using Router ${window.CONTRACTS.router} to swap ${window.TOKENS.UserToken} → ${window.TOKENS.TargetToken}`);

    // Simulate profit/loss
    const profit = (Math.random() - 0.5) * 10;
    const newBalance = client.current_balance + profit;
    await updateClientBalance(userid, newBalance);

    logWithTime(`[Result] Cycle for ${client.userid} — Profit ${profit.toFixed(2)} USDT`);
}
