// chart.txt — Chart.js balance visualization
logToPanel("chart.txt loaded");

function initChart() {
    const canvas = document.getElementById("balanceChart");
    if (!canvas) {
        logToPanel("ERROR: balanceChart canvas not found");
        return;
    }

    const ctx = canvas.getContext("2d");

    // Seed chart with dummy data so it's visible immediately
    const startBalance = parseFloat(document.getElementById("investBalance").value) || 500;

    window.balanceChart = new Chart(ctx, {
        type: "line",
        data: {
            labels: ["Init"],
            datasets: [{
                label: "Investment Balance",
                data: [startBalance], // seed with current invest balance
                borderColor: "#f3ba2f",
                backgroundColor: "rgba(243,186,47,0.2)",
                fill: true,
                tension: 0.3
            }]
        },
        options: {
            responsive: false,
            scales: {
                x: { title: { display: true, text: "Cycle" } },
                y: { title: { display: true, text: "Balance (USDT)" } }
            },
            plugins: {
                legend: { labels: { color: "white" } }
            }
        }
    });

    logToPanel("Chart initialized successfully with dummy seed");
}

function updateChart(balance) {
    if (!window.balanceChart) {
        logToPanel("Chart not initialized — calling initChart()");
        initChart();
    }

    const chart = window.balanceChart;

    // Ensure dataset exists
    if (!chart.data.datasets || chart.data.datasets.length === 0) {
        chart.data.datasets = [{
            label: "Investment Balance",
            data: [],
            borderColor: "#f3ba2f",
            backgroundColor: "rgba(243,186,47,0.2)",
            fill: true,
            tension: 0.3
        }];
    }

    const cycle = chart.data.labels.length + 1;
    chart.data.labels.push(`Cycle ${cycle}`);
    chart.data.datasets[0].data.push(balance);
    chart.update();
}

window.initChart = initChart;
window.updateChart = updateChart;

// Initialize chart once window is fully loaded
window.addEventListener("load", () => {
    logToPanel("initChart triggered on window load");
    initChart();
});
