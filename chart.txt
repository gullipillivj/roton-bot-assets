// chart.txt — Chart.js balance visualization helpers
logToPanel("chart.txt loaded");

function updateChart(balance, profit = null) {
    if (!window.balanceChart) {
        logToPanel("ERROR: balanceChart not found — chart must be initialized in HTML");
        return;
    }

    const chart = window.balanceChart;
    const cycle = chart.data.labels.length + 1;
    chart.data.labels.push(`Cycle ${cycle}`);

    // Balance dataset
    chart.data.datasets[0].data.push(balance);

    // Profit dataset if present
    if (profit !== null && chart.data.datasets[1]) {
        chart.data.datasets[1].data.push(profit);
    }

    chart.update();
}

window.updateChart = updateChart;
