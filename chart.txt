// chart.txt — Chart.js balance visualization helpers (Bar Graph)
logToPanel("chart.txt loaded");

function updateChart(balance, profit = null) {
    const chart = window.balanceChart;
    if (!chart) {
        logToPanel("ERROR: balanceChart not found — chart must be initialized in HTML");
        return;
    }

    // Add next cycle label
    const cycle = chart.data.labels.length + 1;
    chart.data.labels.push(`Cycle ${cycle}`);

    // Append new data points cumulatively
    chart.data.datasets[0].data.push(balance);
    if (profit !== null && chart.data.datasets[1]) {
        chart.data.datasets[1].data.push(profit);
    }

    chart.update();
}

// Switch chart type to bar and style datasets
if (window.balanceChart) {
    window.balanceChart.config.type = "bar";
    window.balanceChart.data.datasets[0].backgroundColor = "rgba(243, 186, 47, 0.7)"; // yellow bars
    window.balanceChart.data.datasets[1].backgroundColor = "rgba(0, 255, 0, 0.7)";   // green bars
    window.balanceChart.data.datasets[0].borderWidth = 1;
    window.balanceChart.data.datasets[1].borderWidth = 1;
    window.balanceChart.update();
}

window.updateChart = updateChart;
