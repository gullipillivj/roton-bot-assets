// contracts.txt — Loader for centralized Brownie contract addresses
logToPanel("contracts.txt loaded");

// Load contracts + tokens + settings from contracts.json
async function loadContractsJSON() {
    try {
        const res = await fetch("/contracts.json");
        const data = await res.json();

        // Expose globally
        window.CONTRACTS = data.contracts;
        window.TOKENS = data.tokens;
        window.CONFIG = data.settings;

        logWithTime("[JSON] contracts.json loaded successfully");
        return data;
    } catch (err) {
        logWithTime("[ERROR] Failed to load contracts.json: " + err);
        return null;
    }
}

// Call loader immediately on startup
loadContractsJSON();
