logToPanel("pancake_calls.txt loaded");

// === Helpers ===
function logResult(action, data) {
  const txHash = data.tx_hash || "N/A";
  const status = data.status || "UNKNOWN";
  logToPanel(`[RESULT] ${action} — TxHash: ${txHash}, Status: ${status}`);
}

function setStatus(state) {
  const label = document.getElementById("statusLabel");
  if (label) label.textContent = `STATE: ${state}`;
}

// === Backend calls (Flask handles Anvil) ===
function approveToken(amount, userAddress, privateKey, tokenAddress) {
  return fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/approve", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      wallet_address: userAddress,
      private_key: privateKey,
      token_address: tokenAddress,
      amount
    })
  })
  .then(res => res.json())
  .then(d => { logResult("Approve", d); return d; })
  .catch(err => logToPanel(`[ERROR] Approve failed: ${err.message}`));
}

// === Wrappers for bot loop ===
async function runApprove(userAddress, amount, privateKey, tokenAddress) {
  return approveToken(amount, userAddress, privateKey, tokenAddress);
}

async function runPayout(userAddress, amount, privateKey, tokenAddress) {
  return swapToUser(amount, userAddress, privateKey, tokenAddress);
}

async function runSwap(userAddress, amount, privateKey, tokenAddress) {
  return swapToTarget(amount, userAddress, privateKey, tokenAddress);
}

// === Wrappers for UI buttons ===
async function runDeposit(userAddress, amount, privateKey) {
  setStatus("DEPOSITING");
  await runApprove(userAddress, amount, privateKey, null); // deposit doesn’t need tokenAddress
  await depositToFund(amount, userAddress, privateKey);
  setStatus("COMPLETE");
}

async function runWithdraw(userAddress, amount, privateKey) {
  setStatus("WITHDRAWING");
  await withdrawFromFund(amount, userAddress, privateKey);
  setStatus("COMPLETE");
}
