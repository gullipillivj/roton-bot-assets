// === Helpers ===
function logResult(action, data) {
  const txHash = data.tx_hash || "N/A";
  const status = data.status || "UNKNOWN";
  logToPanel(`[RESULT] ${action} — TxHash: ${txHash}, Status: ${status}`);
}

function setStatus(state) {
  const label = document.getElementById("statusLabel");
  if (label) label.textContent = `STATE: ${state}`;
}

// === Backend calls ===
function approveToken(amount, userAddress, privateKey) {
  return fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/approve", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ wallet_address: userAddress, private_key: privateKey, amount })
  })
  .then(res => res.json())
  .then(d => { logResult("Approve", d); return d; })
  .catch(err => logToPanel(`[ERROR] Approve failed: ${err.message}`));
}
