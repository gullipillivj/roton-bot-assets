// pancake_calls.txt — PancakeSwap + SolidarityFund interaction layer
logToPanel("pancake_calls.txt loaded");

const web3 = new Web3("http://127.0.0.1:5050");
window.web3 = web3;

const userTokenAddress = "0x5FC8d32690cc91D4c39d9d3abcBD16989F875707";
const targetTokenAddress = "0x0165878A594ca255338adfa4d48449f69242Eb8F";
const routerAddress      = "0xa513E6E4b8f2a923D98304ec87F64353C4D5C853";
const solidarityFundAddress = "0x2279B7A0a67DB372996a5FaB50D91eAA73d2eBe6";

const routerABI = [ /* Pancake Router ABI JSON here */ ];
const erc20ABI = [
  { "constant": false, "inputs":[{"name":"spender","type":"address"},{"name":"value","type":"uint256"}], "name":"approve", "outputs":[{"name":"","type":"bool"}], "type":"function" },
  { "constant": true, "inputs":[{"name":"owner","type":"address"}], "name":"balanceOf", "outputs":[{"name":"balance","type":"uint256"}], "type":"function" }
];
const fundABI = [ /* SolidarityFund ABI JSON here */ ];

const router = new web3.eth.Contract(routerABI, routerAddress);
const fund   = new web3.eth.Contract(fundABI, solidarityFundAddress);

function setStatus(state) {
  const label = document.getElementById("statusLabel");
  if (label) label.textContent = `STATE: ${state}`;
}

function logReceipt(action, receipt) {
  const txHash = receipt.transactionHash || "N/A";
  const block  = receipt.blockNumber || "N/A";
  const gas    = receipt.gasUsed || "N/A";
  const status = receipt.status ? "SUCCESS" : "FAILED";
  logToPanel(`[RESULT] ${action} — TxHash: ${txHash}, Block: ${block}, Gas: ${gas}, Status: ${status}`);
}

// === Functions with PromiEvent logging ===
function approveToken(token, amount, user) {
  logToPanel(`[DEBUG] Approving ${amount} of ${token} for ${user}`);
  const tokenContract = new web3.eth.Contract(erc20ABI, token);
  return tokenContract.methods.approve(routerAddress, amount).send({ from: user })
    .on('transactionHash', hash => logToPanel(`[INFO] Approve submitted: ${hash}`))
    .on('receipt', receipt => logReceipt("Approve", receipt))
    .on('error', err => logToPanel(`[ERROR] Approve failed: ${err.message}`));
}

function depositToFund(token, amount, user) {
  logToPanel(`[DEBUG] Depositing ${amount} of ${token} into fund for ${user}`);
  return fund.methods.deposit(token, amount).send({ from: user })
    .on('transactionHash', hash => logToPanel(`[INFO] Deposit submitted: ${hash}`))
    .on('receipt', receipt => logReceipt("Deposit", receipt))
    .on('error', err => logToPanel(`[ERROR] Deposit failed: ${err.message}`));
}

function withdrawFromFund(token, amount, user) {
  logToPanel(`[DEBUG] Withdrawing ${amount} of ${token} from fund for ${user}`);
  return fund.methods.withdraw(token, amount).send({ from: user })
    .on('transactionHash', hash => logToPanel(`[INFO] Withdraw submitted: ${hash}`))
    .on('receipt', receipt => logReceipt("Withdraw", receipt))
    .on('error', err => logToPanel(`[ERROR] Withdraw failed: ${err.message}`));
}

function swapToTarget(amount, user) {
  logToPanel(`[DEBUG] Swapping ${amount} ${userTokenAddress} -> ${targetTokenAddress} for ${user}`);
  const path = [userTokenAddress, targetTokenAddress];
  const deadline = Math.floor(Date.now() / 1000) + 600;
  return router.methods.swapExactTokensForTokens(amount, 0, path, user, deadline).send({ from: user })
    .on('transactionHash', hash => logToPanel(`[INFO] SwapToTarget submitted: ${hash}`))
    .on('receipt', receipt => logReceipt("SwapToTarget", receipt))
    .on('error', err => logToPanel(`[ERROR] SwapToTarget failed: ${err.message}`));
}

function swapToUser(amount, user) {
  logToPanel(`[DEBUG] Swapping ${amount} ${targetTokenAddress} -> ${userTokenAddress} for ${user}`);
  const path = [targetTokenAddress, userTokenAddress];
  const deadline = Math.floor(Date.now() / 1000) + 600;
  return router.methods.swapExactTokensForTokens(amount, 0, path, user, deadline).send({ from: user })
    .on('transactionHash', hash => logToPanel(`[INFO] SwapToUser submitted: ${hash}`))
    .on('receipt', receipt => logReceipt("SwapToUser", receipt))
    .on('error', err => logToPanel(`[ERROR] SwapToUser failed: ${err.message}`));
}

// === Wrappers for UI buttons ===
async function runDeposit(username, amount) {
  const currentUser = userAddressMap[username];
  if (!currentUser) {
    logToPanel("[ERROR] Unknown user: " + username);
    return;
  }
  setStatus("DEPOSITING");
  logToPanel(`[INFO] Starting deposit for ${username}, amount=${amount}`);
  await approveToken(userTokenAddress, amount, currentUser);
  await depositToFund(userTokenAddress, amount, currentUser);
  setStatus("COMPLETE");
}

async function runWithdraw(username, amount) {
  const currentUser = userAddressMap[username];
  if (!currentUser) {
    logToPanel("[ERROR] Unknown user: " + username);
    return;
  }
  setStatus("WITHDRAWING");
  logToPanel(`[INFO] Starting withdraw for ${username}, amount=${amount}`);
  await withdrawFromFund(userTokenAddress, amount, currentUser);
  setStatus("COMPLETE");
}

// === Expose globally ===
window.userTokenAddress   = userTokenAddress;
window.targetTokenAddress = targetTokenAddress;
window.fund               = fund;
window.approveToken       = approveToken;
window.depositToFund      = depositToFund;
window.withdrawFromFund   = withdrawFromFund;
window.swapToTarget       = swapToTarget;
window.swapToUser         = swapToUser;
window.runDeposit         = runDeposit;
window.runWithdraw        = runWithdraw;
window.setStatus          = setStatus;
