// === Helpers ===
function setStatus(state) {
  const label = document.getElementById("statusLabel");
  if (label) label.textContent = `STATE: ${state}`;
}

function logResult(action, data) {
  const txHash = data.tx_hash || "N/A";
  const status = data.status || "UNKNOWN";
  logToPanel(`[RESULT] ${action} — TxHash: ${txHash}, Status: ${status}`);
}

// === Backend calls ===
function approveToken(token, amount, userAddress) {
  logToPanel(`[INFO] Approving ${amount} of ${token} for ${userAddress}`);
  return fetch("https://almanac-sierra-rundown.ngrok-free.dev/runbot/approve", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ wallet_address: userAddress, token, amount })
  })
  .then(res => res.json())
  .then(d => { logResult("Approve", d); return d; })
  .catch(err => logToPanel(`[ERROR] Approve failed: ${err.message}`));
}

// === Wrappers for bot loop compatibility ===
async function runApprove(userAddress, amount) {
  return approveToken(window.userTokenAddress, amount, userAddress);
}

async function runSwap(userAddress, amount) {
  return swapToTarget(amount, userAddress);
}

async function runPayout(userAddress, amount) {
  return swapToUser(amount, userAddress);
}

// === Wrappers for UI buttons ===
async function runDeposit(userAddress, amount) {
  setStatus("DEPOSITING");
  await runApprove(userAddress, amount);
  await depositToFund(window.userTokenAddress, amount, userAddress);
  setStatus("COMPLETE");
}

async function runWithdraw(userAddress, amount) {
  setStatus("WITHDRAWING");
  await withdrawFromFund(window.userTokenAddress, amount, userAddress);
  setStatus("COMPLETE");
}
