logToPanel("pancake_calls.txt loaded");

function safeParse(text) {
  try { return JSON.parse(text); } catch { return null; }
}
function logResult(action, data) {
  if (data && data.tx_hash) {
    logToPanel(`[RESULT] ${action} — TxHash=${data.tx_hash}, Status=${data.status || "UNKNOWN"}`);
  } else {
    logToPanel(`[RESULT] ${action} — Raw=${JSON.stringify(data)}`);
  }
}

// === Backend calls ===
async function approveToken(amount, userAddress, privateKey) {
  try {
    const res = await fetch(`${BASE_URL}/runbot/approve`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ wallet_address: userAddress, private_key: privateKey, amount })
    });
    const parsed = safeParse(await res.text());
    logResult("Approve", parsed);
    return parsed;
  } catch (err) {
    logToPanel(`[ERROR] Approve failed: ${err.message}`);
  }
}
