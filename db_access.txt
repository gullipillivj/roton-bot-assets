// db_access.txt — Client database access
logToPanel("db_access.txt loaded");

const CLIENTS_URL = "https://almanac-sierra-rundown.ngrok-free.dev/clients";

// --- Connection: fetch raw JSON ---
async function connectClients() {
    try {
        const res = await fetch(CLIENTS_URL);
        const text = await res.text();
        logWithTime("[DEBUG] Raw response from /clients:");
        logWithTime(text);
        return text; // raw string
    } catch (err) {
        logWithTime(`[ERROR] connectClients() failed: ${err}`);
        return null;
    }
}

// --- Parser: extract one client by userid or name ---
async function parseClient(userid) {
    try {
        const raw = await connectClients();
        if (!raw) return null;

        const data = JSON.parse(raw);
        const client = data.clients.find(c => c.userid === userid || c.name === userid);
        if (!client) {
            logWithTime(`[ERROR] Client ${userid} not found`);
            return null;
        }
        logWithTime(`[DB] Client ${userid} parsed successfully`);
        // Set active user globally
        window.activeUserId = client.userid;
        return client;
    } catch (err) {
        logWithTime(`[ERROR] parseClient(${userid}) failed: ${err}`);
        return null;
    }
}

// --- Deposit ---
async function deposit(userid, amount) {
    try {
        const res = await fetch(`${CLIENTS_URL}/deposit/${userid}/${amount}`);
        return await res.json();
    } catch (err) {
        logWithTime(`[ERROR] deposit(${userid}) failed: ${err}`);
        return null;
    }
}

// --- Withdraw ---
async function withdraw(userid, amount) {
    try {
        const res = await fetch(`${CLIENTS_URL}/withdraw/${userid}/${amount}`);
        return await res.json();
    } catch (err) {
        logWithTime(`[ERROR] withdraw(${userid}) failed: ${err}`);
        return null;
    }
}

// === Expose globally ===
window.connectClients = connectClients;
window.parseClient    = parseClient;
window.deposit        = deposit;
window.withdraw       = withdraw;
