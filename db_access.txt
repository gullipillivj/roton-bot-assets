// db_access.txt

const API_BASE = "http://localhost:5003"; // or your ngrok URL pointing to port 5003

// --- Fetch one client ---
async function getClientInfo(userid) {
    try {
        const res = await fetch(`${API_BASE}/api/client/${userid}`);
        const data = await res.json();   // no manual JSON.parse needed
        if (data.error) {
            logWithTime(`[ERROR] Client ${userid} not found`);
            return null;
        }
        logWithTime(`[DB] Client ${userid} info loaded`);
        return data;
    } catch (err) {
        logWithTime(`[ERROR] getClientInfo(${userid}) failed: ${err}`);
        return null;
    }
}

// --- Fetch all clients ---
async function getAllClients() {
    try {
        const res = await fetch(`${API_BASE}/api/clients`);
        const data = await res.json();
        logWithTime(`[DB] All clients loaded (${data.clients.length})`);
        return data.clients;
    } catch (err) {
        logWithTime(`[ERROR] getAllClients() failed: ${err}`);
        return [];
    }
}
