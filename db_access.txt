const CLIENTS_URL = "https://almanac-sierra-rundown.ngrok-free.dev/clients";

// --- Fetch all clients (debug: show raw response only) ---
async function getAllClients() {
    try {
        const res = await fetch(CLIENTS_URL);
        const text = await res.text(); // get raw response as plain text
        logWithTime("[DEBUG] Raw response from /clients:");
        logWithTime(text);             // dump raw JSON/HTML string into log panel
        // Do NOT parse for now
        return text;
    } catch (err) {
        logWithTime(`[ERROR] getAllClients() failed: ${err}`);
        return null;
    }
}

// --- Fetch one client by userid or name (debug: just show all) ---
async function getClientInfo(userid) {
    const raw = await getAllClients();
    logWithTime(`[DEBUG] getClientInfo called for ${userid}`);
    // No parsing yet — just return raw string
    return raw;
}
