// buttons.txt

// --- Helper: update UI labels from client data ---
function updateUIFromClient(client) {
    if (!client) return;
    // Update user label
    document.getElementById("currentUserLabel").textContent = `Current User: ${client.name}`;
    // Optionally show balance or amount
    document.getElementById("amountBox").value = client.current_balance;
    logWithTime(`[UI] Updated label and amount for ${client.name}`);
}

// --- Deposit button ---
document.getElementById("depositBtn").addEventListener("click", async () => {
    const amount = parseFloat(document.getElementById("amountBox").value);
    if (!amount || amount <= 0) {
        logWithTime("[Deposit] Invalid amount");
        return;
    }

    const result = await deposit("mockuser1", amount);
    if (result) {
        const client = await parseClient("mockuser1");
        if (client) {
            setActiveUser(client.userid, client.name);
            updateUIFromClient(client); // update label + amount box
            logWithTime(`[Deposit] ${amount} USDT deposited. New balance: ${client.current_balance} USDT`);
        }
    }
});

// --- Withdraw button ---
document.getElementById("withdrawBtn").addEventListener("click", async () => {
    const amount = parseFloat(document.getElementById("amountBox").value);
    if (!amount || amount <= 0) {
        logWithTime("[Withdraw] Invalid amount");
        return;
    }

    const clientBefore = await parseClient("mockuser1");
    if (!clientBefore || amount > clientBefore.current_balance) {
        logWithTime("[Withdraw] Amount exceeds current balance");
        return;
    }

    const result = await withdraw("mockuser1", amount);
    if (result) {
        const client = await parseClient("mockuser1");
        if (client) {
            setActiveUser(client.userid, client.name);
            updateUIFromClient(client); // update label + amount box
            logWithTime(`[Withdraw] ${amount} USDT withdrawn. New balance: ${client.current_balance} USDT`);
        }
    }
});
