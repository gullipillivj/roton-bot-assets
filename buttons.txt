// buttons.txt

// --- Deposit button ---
document.getElementById("depositBtn").addEventListener("click", async () => {
    const amount = parseFloat(document.getElementById("amountBox").value);
    if (!amount || amount <= 0) {
        logWithTime("[Deposit] Invalid amount");
        return;
    }

    const result = await deposit("mockuser1", amount); // force mockuser1
    if (result) {
        const client = await parseClient("mockuser1");
        if (client) {
            setActiveUser(client.userid, client.name); // update label/user name
            logWithTime(`[Deposit] ${amount} USDT deposited. New balance: ${client.current_balance} USDT`);
        }
    }
});

// --- Withdraw button ---
document.getElementById("withdrawBtn").addEventListener("click", async () => {
    const amount = parseFloat(document.getElementById("amountBox").value);
    if (!amount || amount <= 0) {
        logWithTime("[Withdraw] Invalid amount");
        return;
    }

    const clientBefore = await parseClient("mockuser1");
    if (!clientBefore || amount > clientBefore.current_balance) {
        logWithTime("[Withdraw] Amount exceeds current balance");
        return;
    }

    const result = await withdraw("mockuser1", amount); // force mockuser1
    if (result) {
        const client = await parseClient("mockuser1");
        if (client) {
            setActiveUser(client.userid, client.name); // update label/user name
            logWithTime(`[Withdraw] ${amount} USDT withdrawn. New balance: ${client.current_balance} USDT`);
        }
    }
});
