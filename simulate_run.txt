// simulate_run.txt — orchestrator
logToPanel("simulate_run.txt loaded");

const { spawn } = require("child_process");

let initialBalance = 0;

// --- Helper: run coins.txt directly ---
async function getBTCPrice() {
    return new Promise((resolve, reject) => {
        const process = spawn("python3", ["coins.txt"]);
        let output = "";
        process.stdout.on("data", data => output += data.toString());
        process.on("close", code => {
            if (code === 0) {
                const match = output.match(/([\d.]+)/);
                resolve(match ? parseFloat(match[1]) : null);
            } else {
                reject(new Error("coins.txt failed"));
            }
        });
    });
}

// --- Single trading cycle ---
async function simulateCycle(cycleNum) {
    logWithTime(`[Cycle ${cycleNum}] started`);

    let balanceBox = document.getElementById("amountBox");
    let balance = parseFloat(balanceBox.value);

    if (!balance || balance <= 0) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — balance is zero`);
        return;
    }

    const price = await getBTCPrice();
    if (!price) {
        logWithTime(`[Cycle ${cycleNum}] No BTC price available`);
        return;
    }

    logWithTime(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycleNum}: Bought BTCUSDT at ${price} USDT`);

    // Simple profit/loss mock
    const change = (Math.random() - 0.5) * 0.2;
    const result = price * change;

    if (result >= 0) {
        logWithTime(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycleNum}: Result Profit (${result.toFixed(2)} USDT)`);
    } else {
        logWithTime(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycleNum}: Result Loss (${result.toFixed(2)} USDT)`);
    }

    balanceBox.value = balance + result;
    updateChart(cycleNum, balanceBox.value);

    logWithTime(`[Cycle ${cycleNum}] completed`);
}

// --- Start bot ---
function initBot(cycles) {
    let balanceBox = document.getElementById("amountBox");
    initialBalance = parseFloat(balanceBox.value) || 0;

    let cycleNum = 1;
    logWithTime(`[Bot] Starting for ${cycles} cycles with initial balance ${initialBalance} USDT`);

    window.botInterval = setInterval(async () => {
        if (cycleNum > cycles) {
            stopBot();
            return;
        }
        await simulateCycle(cycleNum);
        cycleNum++;
    }, 30000); // 30 seconds
}

// --- Stop bot ---
function stopBot() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        logWithTime("[Bot] Stopped");
    }
}

// Expose globally
window.initBot = initBot;
window.stopBot = stopBot;
