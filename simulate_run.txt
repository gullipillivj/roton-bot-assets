// simulate_run.txt — Bot cycle simulation with PancakeSwap calls
logToPanel("simulate_run.txt loaded");

function setStatus(state) {
  const label = document.getElementById("statusLabel");
  if (label) label.textContent = `STATE: ${state}`;
}

function logReceipt(action, receipt) {
  const txHash = receipt.transactionHash || "N/A";
  const block  = receipt.blockNumber || "N/A";
  const gas    = receipt.gasUsed || "N/A";
  const status = receipt.status ? "SUCCESS" : "FAILED";
  logToPanel(`[RESULT] ${action} — TxHash: ${txHash}, Block: ${block}, Gas: ${gas}, Status: ${status}`);
}

function stopBot() {
  if (window.botInterval) {
    clearInterval(window.botInterval);
    window.botInterval = null;
    setStatus("STOPPED");
    logToPanel("[INFO] Bot stopped");
  } else {
    logToPanel("[WARN] Bot was not running");
  }
}

let currentCycle = 1;
let cycleTimer = null;

async function simulateCycle(cycle) {
  logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);

  const coin = await pickRandomCoin();
  const stats = await window.getBinanceStats(coin);

  if (!stats) {
    logToPanel(`[WARN] Cycle ${cycle}: No stats for ${coin}, skipping`);
    return;
  }

  const buyPrice = parseFloat(stats.lastPrice || 0);
  if (isNaN(buyPrice) || buyPrice <= 0) {
    logToPanel(`[WARN] Cycle ${cycle}: Invalid price for ${coin}, skipping`);
    return;
  }

  logToPanel(`[INFO] Cycle ${cycle}: Bought ${coin} at ${buyPrice.toFixed(4)} USDT`);

  // === PancakeSwap interaction (Flask backend calls) ===
  swapToTarget(buyPrice, window.userAddress, window.privateKey)
    .then(d => {
      logToPanel(`[INFO] Cycle ${cycle}: SwapToTarget TxHash=${d.tx_hash || "N/A"}`);
    })
    .catch(err => logToPanel(`[ERROR] Cycle ${cycle}: SwapToTarget failed: ${err.message}`));

  swapToUser(buyPrice, window.userAddress, window.privateKey)
    .then(d => {
      logToPanel(`[INFO] Cycle ${cycle}: SwapToUser TxHash=${d.tx_hash || "N/A"}`);
    })
    .catch(err => logToPanel(`[ERROR] Cycle ${cycle}: SwapToUser failed: ${err.message}`));

  // === Profit/loss calculation ===
  const percentChange = parseFloat(stats.priceChangePercent || 0);
  const grossPnL = calculatePnL(buyPrice, percentChange);

  // === Deduct gas + Pancake fees using coins.txt helpers ===
  const tradeCost = await window.calculateTradeCost(coin, buyPrice, "bsc");
  const netPnL = grossPnL - (tradeCost.totalCost || 0);

  if (netPnL >= 0) {
    logToPanel(`[INFO] Cycle ${cycle}: Net Profit ${netPnL.toFixed(2)} USDT (after fees)`);
  } else {
    logToPanel(`[INFO] Cycle ${cycle}: Net Loss ${netPnL.toFixed(2)} USDT (after fees)`);
  }

  updateChart(buyPrice, netPnL);

  const liveProfitBox = document.getElementById("liveProfitValue");
  if (liveProfitBox) {
    liveProfitBox.textContent = `${netPnL.toFixed(2)} USDT`;
  }

  logToPanel(`[INFO] simulateCycle(${cycle}) complete`);
}

function startSimulation(totalCycles) {
  logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — initBot started for ${totalCycles} cycles (1‑minute interval)`);

  currentCycle = 1;
  cycleTimer = setInterval(async () => {
    if (currentCycle > totalCycles) {
      clearInterval(cycleTimer);
      logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Simulation complete`);
      setStatus("STOPPED");
      return;
    }

    await simulateCycle(currentCycle);
    currentCycle++;
  }, 60000);
}

function initBot(totalCycles = 10) {
  setStatus("RUNNING");
  logToPanel("[INFO] Bot started");
  startSimulation(totalCycles);
}

// === Helpers ===
async function pickRandomCoin() {
  try {
    const pairs = await window.getBinancePairs();
    if (!pairs || pairs.length === 0) {
      logToPanel("[WARN] No live pairs fetched, fallback to BTCUSDT");
      return "BTCUSDT";
    }
    return pairs[Math.floor(Math.random() * pairs.length)];
  } catch (e) {
    logToPanel("[ERROR] pickRandomCoin failed: " + e);
    return "BTCUSDT";
  }
}

function calculatePnL(price, percentChange) {
  if (!percentChange) return 0;
  const changeFactor = percentChange / 100;
  return price * changeFactor;
}

// Expose globally
window.startSimulation = startSimulation;
window.simulateCycle = simulateCycle;
window.initBot = initBot;
window.stopBot = stopBot;
