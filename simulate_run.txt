// simulate_run.txt — orchestrator
logToPanel("simulate_run.txt loaded");

// Run a single trading cycle
async function simulateCycle(userid, cycleNum) {
    logWithTime(`[Cycle ${cycleNum}] started`);

    // Example: load config and execute trade
    const cfg = loadConfigFromUI();
    const result = await executeTrade(userid, cfg);

    if (result && result.profit !== undefined) {
        logWithTime(`[Cycle ${cycleNum}] Profit: ${result.profit.toFixed(2)} USDT`);
        await updateClientBalance(userid, result.newBalance.toFixed(2));
        updateChart(cycleNum, result.newBalance);
    } else {
        logWithTime(`[Cycle ${cycleNum}] Trade failed`);
    }

    logWithTime(`[Cycle ${cycleNum}] completed`);
}

// Start bot for N cycles
function initBot_run(cycles) {
    logWithTime(`[Bot] Starting for ${cycles} cycles`);
    let userid = "demoUser"; // replace with actual user ID
    let cycleNum = 1;

    window.botInterval = setInterval(async () => {
        if (cycleNum > cycles) {
            stopBot_run();
            return;
        }
        await simulateCycle(userid, cycleNum);
        cycleNum++;
    }, CONFIG.cycleInterval || 15000);
}

// Stop bot
function stopBot_run() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        logWithTime("[Bot] Stopped");
    }
}

// Expose globally so main_run.txt can call them
window.initBot_run = initBot_run;
window.stopBot_run = stopBot_run;
