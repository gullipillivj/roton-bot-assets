// --- Helper: call coins.txt backend ---
async function getBestCoin() {
    try {
        // Assume you expose categorize_coins() via an API endpoint or subprocess
        const res = await fetch("/coins/categories"); // backend route to run coins.txt
        const categories = await res.json();

        // Pick best coin from short_term list (stub logic)
        const bestCoin = categories.short_term[0];
        logWithTime(`[CoinSelect] Best coin chosen: ${bestCoin.symbol}`);
        return bestCoin;
    } catch (err) {
        logWithTime("[CoinSelect] Failed to fetch coins:", err);
        return null;
    }
}

    return { status: "success", symbol: tokenSymbol, amount };
}

// --- Single trading cycle ---
async function simulateCycle(userid, cycleNum, amount) {
    logWithTime(`[Cycle ${cycleNum}] started`);

    const client = await parseClient(userid);
    if (!client || client.current_balance <= 0) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — balance is zero`);
        return;
    }

    const bestCoin = await getBestCoin();
    if (!bestCoin) {
        logWithTime(`[Cycle ${cycleNum}] No coin selected`);
        return;
    }

    const result = await pancakeBuy(bestCoin.symbol, amount);
    if (result && result.status === "success") {
        logWithTime(`[Cycle ${cycleNum}] Trade success: ${bestCoin.symbol}`);
        await updateClientBalance(userid, client.current_balance - amount);
        updateChart(cycleNum, client.current_balance - amount);
    } else {
        logWithTime(`[Cycle ${cycleNum}] Trade failed`);
    }

    logWithTime(`[Cycle ${cycleNum}] completed`);
}

// --- Start bot ---
function initBot_run(cycles, amount) {
    logWithTime(`[Bot] Starting for ${cycles} cycles with ${amount} USDT`);
    let userid = "mockuser1";
    let cycleNum = 1;

    window.botInterval = setInterval(async () => {
        if (cycleNum > cycles) {
            stopBot_run();
            return;
        }
        await simulateCycle(userid, cycleNum, amount);
        cycleNum++;
    }, 30000); // 30 seconds
}

// --- Stop bot ---
function stopBot_run() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        logWithTime("[Bot] Stopped");
    }
}

// Expose globally
window.initBot_run = initBot_run;
window.stopBot_run = stopBot_run;
