// simulate_run.txt — orchestrator
logToPanel("simulate_run.txt loaded");

let initialBalance = 0; 
let cycleCount = 0;
let swapCount = 0;
let lastTradeTimestamp = 0;

const HOLDING_PERIOD = 60; // 1 minute
const MAX_CYCLES = 10;
const MAX_SWAPS = 3;
const GAS_COST_USDT = 2; // estimated fee

// --- Import Pancake functions ---
import { approveToken, swapToToken, swapToUSDT } from "./pancake_calls.txt";

// --- Helper: get coins (Binance API later) ---
async function getCoins() {
    return [{ symbol: "BTCUSDT", address: "0xBTCAddress", price: 30000 }];
}

// --- Single trading cycle ---
async function simulateCycle(cycleNum, user) {
    logWithTime(`[Cycle ${cycleNum}] started`);

    let balanceBox = document.getElementById("amountBox");
    let balance = parseFloat(balanceBox.value);

    if (!balance || balance <= 0) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — balance is zero`);
        return;
    }

    // Respect holding period
    const now = Math.floor(Date.now() / 1000);
    if (now < lastTradeTimestamp + HOLDING_PERIOD) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — holding period active`);
        return;
    }

    // Stop if max swaps reached
    if (swapCount >= MAX_SWAPS) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — max swaps reached`);
        return;
    }

    const coins = await getCoins();
    const chosenCoin = coins[0]; // BTC for now

    logWithTime(`[INFO] Cycle ${cycleNum}: Evaluating ${chosenCoin.symbol} at ${chosenCoin.price} USDT`);

    const profit = balance - initialBalance;
    if (profit > GAS_COST_USDT) {
        // Approve USDT for Pancake router
        await approveToken("0xUSDTAddress", balance, user);

        // Swap USDT -> Coin
        await swapToToken("0xUSDTAddress", chosenCoin.address, balance, user);

        balanceBox.value = balance; // update UI
        logWithTime(`[Cycle ${cycleNum}] Executed swap to ${chosenCoin.symbol}`);

        swapCount++;
        lastTradeTimestamp = now;
    } else {
        logWithTime(`[Cycle ${cycleNum}] No profitable swap, holding coin.`);
    }

    cycleCount++;
    logWithTime(`[Cycle ${cycleNum}] completed`);
}

// --- Start bot ---
function initBot_run(cycles, user) {
    let balanceBox = document.getElementById("amountBox");
    initialBalance = parseFloat(balanceBox.value) || 0;

    let cycleNum = 1;
    logWithTime(`[Bot] Starting for ${cycles} cycles with initial balance ${initialBalance} USDT`);

    window.botInterval = setInterval(async () => {
        if (cycleNum > cycles || cycleCount >= MAX_CYCLES) {
            stopBot_run();
            return;
        }
        await simulateCycle(cycleNum, user);
        cycleNum++;
    }, 60000); // 1 minute
}

// --- Stop bot ---
function stopBot_run() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        logWithTime("[Bot] Stopped");
    }
}

window.initBot_run = initBot_run;
window.stopBot_run = stopBot_run;
