// simulate_run.txt — orchestrator
logToPanel("simulate_run.txt loaded");

// --- Get best coin from coins.txt backend ---
async function getBestCoin() {
    try {
        const res = await fetch("/coins/categories"); // backend route to run coins.txt
        const categories = await res.json();
        const bestCoin = categories.short_term[0]; 
        logWithTime(`[CoinSelect] Best coin chosen: ${bestCoin.symbol}`);
        return bestCoin;
    } catch (err) {
        logWithTime("[CoinSelect] Failed to fetch coins:", err);
        return null;
    }
}

// --- PancakeSwap buy (real Web3 call) ---
async function pancakeBuy(tokenAddress, amount) {
    try {
        const tx = await pancakeRouter.methods.swapExactTokensForTokens(
            amount,
            0, // minOut
            [USDT_ADDRESS, tokenAddress],
            USER_WALLET,
            Math.floor(Date.now() / 1000) + 60 * 10 // deadline
        ).send({ from: USER_WALLET });

        const gasUsed = tx.gasUsed * tx.effectiveGasPrice;
        logWithTime(`[PancakeSwap] Trade success: ${amount} USDT -> ${tokenAddress}, gas: ${gasUsed}`);
        return { status: "success", spent: amount, gas: gasUsed };
    } catch (err) {
        logWithTime("[PancakeSwap] Trade failed:", err);
        return { status: "failed" };
    }
}

// --- Single trading cycle ---
async function simulateCycle(cycleNum) {
    logWithTime(`[Cycle ${cycleNum}] started`);

    let balanceBox = document.getElementById("amountBox");
    let balance = parseFloat(balanceBox.value);

    if (!balance || balance <= 0) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — balance is zero`);
        return;
    }

    const bestCoin = await getBestCoin();
    if (!bestCoin) {
        logWithTime(`[Cycle ${cycleNum}] No coin selected`);
        return;
    }

    const result = await pancakeBuy(bestCoin.address, balance);
    if (result.status === "success") {
        balance -= (result.spent + result.gas); // deduct actual values
        balanceBox.value = balance;
        logWithTime(`[Cycle ${cycleNum}] New balance: ${balance} USDT`);
        updateChart(cycleNum, balance);
    }

    logWithTime(`[Cycle ${cycleNum}] completed`);
}

// --- Start bot ---
function initBot_run(cycles) {
    let cycleNum = 1;
    logWithTime(`[Bot] Starting for ${cycles} cycles`);

    window.botInterval = setInterval(async () => {
        if (cycleNum > cycles) {
            stopBot_run();
            return;
        }
        await simulateCycle(cycleNum);
        cycleNum++;
    }, 30000); // 30 seconds
}

// --- Stop bot ---
function stopBot_run() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        logWithTime("[Bot] Stopped");
    }
}

// Expose globally
window.initBot_run = initBot_run;
window.stopBot_run = stopBot_run;
