logToPanel("simulate_run.txt loaded");


function stopBot() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        window.botInterval = null;
        logToPanel("[INFO] Bot stopped");
    } else {
        logToPanel("[WARN] Bot was not running");
    }
}




let currentCycle = 1;
let cycleTimer = null;

async function simulateCycle(cycle) {
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);

    // Pick coin from Binance pairs
    const coin = await pickRandomCoin();
    const stats = await window.getBinanceStats(coin);

    if (!stats) {
        logToPanel(`[WARN] Cycle ${cycle}: No stats for ${coin}, skipping`);
        return;
    }

    const buyPrice = parseFloat(stats.lastPrice || 0);
    if (isNaN(buyPrice) || buyPrice <= 0) {
        logToPanel(`[WARN] Cycle ${cycle}: Invalid price for ${coin}, skipping`);
        return;
    }

    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Bought ${coin} at ${buyPrice.toFixed(4)} USDT`);

    // Profit/loss based on actual 24hr % change
    const percentChange = parseFloat(stats.priceChangePercent || 0);
    const result = calculatePnL(buyPrice, percentChange);

    if (result >= 0) {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Profit ${result.toFixed(2)} USDT`);
    } else {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Loss ${result.toFixed(2)} USDT`);
    }

    // Update chart with valid values
    updateChart(buyPrice, result);

    // Update live profit box
    const liveProfitBox = document.getElementById("liveProfitValue");
    if (liveProfitBox) {
        liveProfitBox.textContent = `${result.toFixed(2)} USDT`;
    }

    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) complete`);
}

function startSimulation(totalCycles) {
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — initBot started for ${totalCycles} cycles (1‑minute interval)`);

    currentCycle = 1;
    cycleTimer = setInterval(async () => {
        if (currentCycle > totalCycles) {
            clearInterval(cycleTimer);
            logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Simulation complete`);
            return;
        }

        await simulateCycle(currentCycle);
        currentCycle++;
    }, 60000);
}

function initBot(totalCycles = 10) {
    startSimulation(totalCycles);
}

// === Helpers ===
async function pickRandomCoin() {
    try {
        const pairs = await window.getBinancePairs();
        if (!pairs || pairs.length === 0) {
            logToPanel("[WARN] No live pairs fetched, fallback to BTCUSDT");
            return "BTCUSDT";
        }
        return pairs[Math.floor(Math.random() * pairs.length)];
    } catch (e) {
        logToPanel("[ERROR] pickRandomCoin failed: " + e);
        return "BTCUSDT";
    }
}

function calculatePnL(price, percentChange) {
    if (!percentChange) return 0;
    const changeFactor = percentChange / 100;
    return price * changeFactor;
}

// Expose globally
window.startSimulation = startSimulation;
window.simulateCycle = simulateCycle;
window.initBot = initBot;
window.stopBot = stopBot;
