// simulate_run.txt — Bot cycle simulation with PancakeSwap calls + Registry integration
logToPanel("simulate_run.txt loaded");

window.currentCycle = 1;
window.cycleTimer = null;
let cumulativePnL = 0;

// === Registry contract setup ===
const registryAddress = "0x04C89607413713Ec9775E14b954286519d836FEf";
const registryAbi = [
  {
    "inputs": [],
    "stateMutability": "nonpayable",
    "type": "constructor"
  },
  {
    "inputs": [
      { "internalType": "string", "name": "symbol", "type": "string" }
    ],
    "name": "getTokenAddress",
    "outputs": [
      { "internalType": "address", "name": "", "type": "address" }
    ],
    "stateMutability": "view",
    "type": "function"
  },
  {
    "inputs": [],
    "name": "owner",
    "outputs": [
      { "internalType": "address", "name": "", "type": "address" }
    ],
    "stateMutability": "view",
    "type": "function"
  },
  {
    "inputs": [
      { "internalType": "string", "name": "symbol", "type": "string" },
      { "internalType": "address", "name": "tokenAddr", "type": "address" }
    ],
    "name": "registerToken",
    "outputs": [],
    "stateMutability": "nonpayable",
    "type": "function"
  }
];
const registryContract = new web3.eth.Contract(registryAbi, registryAddress);

async function getTokenAddress(symbol) {
  try {
    const addr = await registryContract.methods.getTokenAddress(symbol).call();
    return addr && addr !== "0x0000000000000000000000000000000000000000" ? addr : null;
  } catch (e) {
    logToPanel("[WARN] Registry lookup failed: " + e.message);
    return null;
  }
}

// === Simulation logic ===
async function simulateCycle(cycle, startBal, profitTargetAbs, stopLossAbs, totalCycles) {
  logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);
  setStatus(`RUNNING cycle ${cycle}/${totalCycles}`);

  const coin = await window.pickRandomCoin();
  logToPanel(`[DEBUG] Picked coin: ${coin}`);

  const stats = await window.getBinanceStats(coin);
  if (!stats) {
    logToPanel(`[WARN] Cycle ${cycle}: No stats for ${coin}, skipping`);
    return 0;
  }

  const buyPrice = parseFloat(stats.lastPrice || 0);
  if (isNaN(buyPrice) || buyPrice <= 0) {
    logToPanel(`[WARN] Cycle ${cycle}: Invalid price for ${coin}, skipping`);
    return 0;
  }

  const investBal = parseFloat(document.getElementById("investBalance").value) || 0;
  const allocation = investBal / totalCycles;
  const coinsBought = allocation / buyPrice;

  logToPanel(`[INFO] Cycle ${cycle}: Bought ${coinsBought.toFixed(4)} ${coin} at ${buyPrice.toFixed(4)} USDT (Allocation ${allocation.toFixed(2)} USDT)`);

  const tokenAddr = await getTokenAddress(coin);

  // === ETH Proxy Fallback for unsupported coins ===
  if (!tokenAddr) {
    logToPanel(`[WARN] Cycle ${cycle}: ${coin} unsupported → converting to ETHUSDT`);

    const ethStats = await window.getBinanceStats("ETHUSDT");
    const ethPrice = parseFloat(ethStats.lastPrice || 0);
    const ethAllocation = allocation / ethPrice;

    const ethAddr = await getTokenAddress("ETHUSDT");
    if (!ethAddr) {
      logToPanel("[ERROR] ETHUSDT address not found in registry, aborting swap");
      return 0;
    }

    try {
      const approveResp = await window.runApprove(USER_ADDR, ethAllocation, USER_PK);
      logToPanel(`[RESULT] Proxy Approve JSON: ${JSON.stringify(approveResp)}`);

      await new Promise(r => setTimeout(r, 2000));

      const swapResp = await window.runSwap(USER_ADDR, ethAllocation, USER_PK);
      logToPanel(`[RESULT] Proxy Swap JSON: ${JSON.stringify(swapResp)}, Converted ${coin} → ${ethAllocation.toFixed(4)} ETH`);
    } catch (err) {
      logToPanel(`[ERROR] Proxy swap failed: ${err.message}`);
    }

    return 0;
  }

  // === Approve + Swap with JSON logs ===
  try {
    const approveResp = await window.runApprove(USER_ADDR, allocation, USER_PK);
    logToPanel(`[RESULT] Approve JSON: ${JSON.stringify(approveResp)}`);

    await new Promise(r => setTimeout(r, 2000));

    const swapResp = await window.runSwap(USER_ADDR, allocation, USER_PK);
    logToPanel(`[RESULT] Swap JSON: ${JSON.stringify(swapResp)}, Coins swapped: ${coinsBought.toFixed(4)} ${coin}`);
  } catch (err) {
    logToPanel(`[ERROR] Swap failed: ${err.message}`);
  }

  const percentChange = parseFloat(stats.priceChangePercent || 0);
  const grossPnL = calculatePnL(allocation, percentChange);
  const tradeCost = await window.calculateTradeCost(coin, allocation, "bsc");
  const netPnL = grossPnL - (tradeCost.totalCost || 0);

  logToPanel(`[INFO] Cycle ${cycle}: Net ${netPnL >= 0 ? "Profit" : "Loss"} ${netPnL.toFixed(2)} USDT (after fees)`);

  updateChart(buyPrice, netPnL);
  const liveProfitBox = document.getElementById("liveProfitValue");
  if (liveProfitBox) liveProfitBox.textContent = `${netPnL.toFixed(2)} USDT`;

  cumulativePnL += netPnL;

  if (cumulativePnL >= profitTargetAbs) {
    logToPanel(`[INFO] Profit target reached after cycle ${cycle}, stopping trading`);
    return "STOP";
  }
  if (cumulativePnL <= -stopLossAbs) {
    logToPanel(`[INFO] Stop loss triggered after cycle ${cycle}, stopping trading`);
    return "STOP";
  }

  logToPanel(`[INFO] simulateCycle(${cycle}) complete`);
  return netPnL;
}

// === Helpers ===
function calculatePnL(amount, percentChange) {
  if (!percentChange) return 0;
  const changeFactor = percentChange / 100;
  return amount * changeFactor;
}

async function pickRandomCoin() {
  try {
    const pairs = await window.getBinancePairs();
    if (!pairs || pairs.length === 0) {
      logToPanel("[WARN] No live pairs fetched, fallback to BTCUSDT");
      return "BTCUSDT";
    }
    const choice = pairs[Math.floor(Math.random() * pairs.length)];
    return choice;
  } catch (e) {
    logToPanel("[ERROR] pickRandomCoin failed: " + e);
    return "BTCUSDT";
  }
}
window.pickRandomCoin = pickRandomCoin;
window.simulateCycle = simulateCycle;
