// === Web3 provider setup ===
// Use VPS IP + /rpc (proxied via nginx)
const web3 = new Web3("http://69.10.54.87/rpc");

// === Registry contract setup ===
const registryAddress = "0x04C89607413713Ec9775E14b954286519d836FEf";
const registryAbi = [ /* same ABI as before */ ];
const registryContract = new web3.eth.Contract(registryAbi, registryAddress);

async function getTokenAddress(symbol) {
  try {
    const addr = await registryContract.methods.getTokenAddress(symbol).call();
    return addr && addr !== "0x0000000000000000000000000000000000000000" ? addr : null;
  } catch (e) {
    logToPanel("[WARN] Registry lookup failed: " + e.message);
    return null;
  }
}

// === Balance initialization ===
async function initBalances(userId) {
  try {
    const res = await fetch(`/runbot/balances`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ wallet_address: USER_ADDR })
    });
    const parsed = JSON.parse(await res.text());
    if (parsed && parsed.startBalance) {
      document.getElementById("startingBalance").value = parsed.startBalance;
      document.getElementById("investBalance").value = parsed.investBalance || parsed.startBalance;
      logToPanel(`[INFO] Initialized balances: Start=${parsed.startBalance}, Invest=${parsed.investBalance}`);
    }
  } catch (err) {
    logToPanel(`[ERROR] Balance init failed: ${err.message}`);
  }
}

// === Approve helper ===
async function runApprove(userAddr, amount, userPk) {
  const account = web3.eth.accounts.privateKeyToAccount(userPk);
  web3.eth.accounts.wallet.add(account);

  const tokenAddr = await getTokenAddress("ETHUSDT"); // fallback
  const tokenContract = new web3.eth.Contract(erc20Abi, tokenAddr);

  const tx = tokenContract.methods.approve(SWAP_ROUTER_ADDR, web3.utils.toWei(amount.toString(), "ether"));
  const gas = await tx.estimateGas({ from: userAddr });
  const data = tx.encodeABI();

  const signedTx = await web3.eth.accounts.signTransaction(
    { to: tokenAddr, data, gas },
    userPk
  );

  return await web3.eth.sendSignedTransaction(signedTx.rawTransaction);
}

// === Swap helper ===
async function runSwap(userAddr, amount, userPk) {
  const account = web3.eth.accounts.privateKeyToAccount(userPk);
  web3.eth.accounts.wallet.add(account);

  const tokenAddr = await getTokenAddress("ETHUSDT"); // fallback
  const router = new web3.eth.Contract(routerAbi, SWAP_ROUTER_ADDR);
  const deadline = Math.floor(Date.now() / 1000) + 60 * 20;

  const tx = router.methods.swapExactTokensForTokens(
    web3.utils.toWei(amount.toString(), "ether"),
    0,
    [tokenAddr, USDT_ADDRESS],
    userAddr,
    deadline
  );

  const gas = await tx.estimateGas({ from: userAddr });
  const data = tx.encodeABI();

  const signedTx = await web3.eth.accounts.signTransaction(
    { to: SWAP_ROUTER_ADDR, data, gas },
    userPk
  );

  return await web3.eth.sendSignedTransaction(signedTx.rawTransaction);
}

window.runApprove = runApprove;
window.runSwap = runSwap;

// === Simulation logic ===
async function simulateCycle(cycle, startBal, profitTargetAbs, stopLossAbs, totalCycles) {
  logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);
  setStatus(`RUNNING cycle ${cycle}/${totalCycles}`);

  const investBal = parseFloat(document.getElementById("investBalance").value) || 0;
  if (investBal <= 0) {
    logToPanel(`[WARN] Cycle ${cycle}: Invest balance is zero, skipping`);
    return 0;
  }

  const allocation = investBal / totalCycles;
  logToPanel(`[INFO] Cycle ${cycle}: Allocation ${allocation.toFixed(2)} USDT`);

  try {
    const approveResp = await window.runApprove(USER_ADDR, allocation, USER_PK);
    logToPanel(`[RESULT] Approve JSON: ${JSON.stringify(approveResp)}`);

    await new Promise(r => setTimeout(r, 2000));

    const swapResp = await window.runSwap(USER_ADDR, allocation, USER_PK);
    logToPanel(`[RESULT] Swap JSON: ${JSON.stringify(swapResp)}`);
  } catch (err) {
    logToPanel(`[ERROR] Swap failed: ${err.message}`);
  }

  logToPanel(`[INFO] simulateCycle(${cycle}) complete`);
  return allocation;
}
