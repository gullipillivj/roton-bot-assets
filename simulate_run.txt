// simulate_run.txt — Bot cycle simulation with PancakeSwap calls
logToPanel("simulate_run.txt loaded");

window.currentCycle = 1;
window.cycleTimer = null;
let cumulativePnL = 0;

async function simulateCycle(cycle, startBal, profitTargetAbs, stopLossAbs, totalCycles) {
  logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);

  // 🔑 Status label update
  setStatus(`RUNNING cycle ${cycle}/${totalCycles}`);

  const coin = await window.pickRandomCoin();
  const stats = await window.getBinanceStats(coin);

  if (!stats) {
    logToPanel(`[WARN] Cycle ${cycle}: No stats for ${coin}, skipping`);
    return 0;
  }

  const buyPrice = parseFloat(stats.lastPrice || 0);
  if (isNaN(buyPrice) || buyPrice <= 0) {
    logToPanel(`[WARN] Cycle ${cycle}: Invalid price for ${coin}, skipping`);
    return 0;
  }

  logToPanel(`[INFO] Cycle ${cycle}: Bought ${coin} at ${buyPrice.toFixed(4)} USDT`);

  const tokenAddr = await window.getTokenAddress(coin);
  if (!tokenAddr) {
    logToPanel(`[WARN] Cycle ${cycle}: No token address for ${coin}, skipping`);
    return 0;
  }

  // === Approve + Swap with JSON logs ===
  const approveResp = await window.runApprove(USER_ADDR, buyPrice, USER_PK);
  logToPanel(`[RESULT] Approve JSON: ${JSON.stringify(approveResp)}`);

  const swapResp = await window.runSwap(USER_ADDR, buyPrice, USER_PK);
  logToPanel(`[RESULT] Swap JSON: ${JSON.stringify(swapResp)}`);

  // === Profit/loss calculation ===
  const percentChange = parseFloat(stats.priceChangePercent || 0);
  const grossPnL = calculatePnL(buyPrice, percentChange);
  const tradeCost = await window.calculateTradeCost(coin, buyPrice, "bsc");
  const netPnL = grossPnL - (tradeCost.totalCost || 0);

  logToPanel(`[INFO] Cycle ${cycle}: Net ${netPnL >= 0 ? "Profit" : "Loss"} ${netPnL.toFixed(2)} USDT (after fees)`);

  updateChart(buyPrice, netPnL);
  const liveProfitBox = document.getElementById("liveProfitValue");
  if (liveProfitBox) liveProfitBox.textContent = `${netPnL.toFixed(2)} USDT`;

  cumulativePnL += netPnL;

  // === Threshold checks ===
  if (cumulativePnL >= profitTargetAbs) {
    logToPanel(`[INFO] Profit target reached after cycle ${cycle}, stopping trading`);
    return "STOP";
  }
  if (cumulativePnL <= -stopLossAbs) {
    logToPanel(`[INFO] Stop loss triggered after cycle ${cycle}, stopping trading`);
    return "STOP";
  }

  logToPanel(`[INFO] simulateCycle(${cycle}) complete`);
  return netPnL;
}

async function startSimulation(totalCycles = 10) {
  const startBal = parseFloat(document.getElementById("startBalance").value) || 0;
  const profitTargetPct = parseFloat(document.getElementById("profitTarget").value) || 0;
  const stopLossPct = parseFloat(document.getElementById("stopLoss").value) || 0;

  const profitTargetAbs = startBal * (profitTargetPct / 100);
  const stopLossAbs = startBal * (stopLossPct / 100);

  logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — initBot started for ${totalCycles} cycles`);

  window.currentCycle = 1;
  window.cycleTimer = setInterval(async () => {
    if (window.currentCycle > totalCycles) {
      clearInterval(window.cycleTimer);
      await finalSettlement();
      return;
    }

    const result = await simulateCycle(window.currentCycle, startBal, profitTargetAbs, stopLossAbs, totalCycles);
    if (result === "STOP") {
      clearInterval(window.cycleTimer);
      await finalSettlement();
      return;
    }

    window.currentCycle++;   // ✅ increment after each run
  }, 60000);
}

async function finalSettlement() {
  logToPanel("[INFO] Final settlement started");

  // Swap holdings back to ETH
  const swapResp = await window.runSwap(USER_ADDR, 0, USER_PK);
  logToPanel(`[RESULT] Final Swap JSON: ${JSON.stringify(swapResp)}`);

  // Payout to user wallet
  const payoutResp = await window.runWithdraw(USER_ADDR, 0, USER_PK);
  logToPanel(`[RESULT] Final Payout JSON: ${JSON.stringify(payoutResp)}`);

  // Refresh balances
  const balances = await window.runBalances(USER_ADDR);
  logToPanel(`[RESULT] Final Balances JSON: ${JSON.stringify(balances)}`);

  if (balances?.fund_token_balance !== undefined) {
    document.getElementById("startBalance").value = balances.fund_token_balance;
    logToPanel(`[INFO] Initial balance updated with P&L: ${balances.fund_token_balance} ETH`);
  }

  // Update clients.db
  await fetch(`${BASE_URL}/updateClientBalance`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ userid: USER_ID, new_balance: balances.fund_token_balance })
  });
  logToPanel("[INFO] Client DB updated with new balance");

  setStatus("STOPPED");
  logToPanel("[INFO] Trading simulation complete");
}

// === Helpers ===
function calculatePnL(price, percentChange) {
  if (!percentChange) return 0;
  const changeFactor = percentChange / 100;
  return price * changeFactor;
}

// === Expose globally ===
window.startSimulation = startSimulation;
window.simulateCycle = simulateCycle;
window.finalSettlement = finalSettlement;
