// simulate_run.txt — Trading simulation with coins.txt
logToPanel("simulate_run.txt loaded");

const { spawn } = require("child_process");

let currentCycle = 1;
let cycleTimer = null;

// --- Helper: run coins.txt directly ---
async function getBTCPrice() {
    return new Promise((resolve, reject) => {
        const process = spawn("python3", ["coins.txt"]);
        let output = "";
        process.stdout.on("data", data => output += data.toString());
        process.on("close", code => {
            if (code === 0) {
                // coins.txt prints "Current BTCUSDT price: 30215.45"
                const match = output.match(/([\d.]+)/);
                resolve(match ? parseFloat(match[1]) : null);
            } else {
                reject(new Error("coins.txt failed"));
            }
        });
    });
}

// --- Single trading cycle ---
async function simulateCycle(cycle) {
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) started`);

    const price = await getBTCPrice();
    if (!price) {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: No BTC price available`);
        return;
    }

    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Bought BTCUSDT at ${price} USDT`);

    // Simple mock profit/loss around ±10%
    const change = (Math.random() - 0.5) * 0.2;
    const result = price * change;

    if (result >= 0) {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Result Profit (${result.toFixed(2)} USDT)`);
    } else {
        logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycle}: Result Loss (${result.toFixed(2)} USDT)`);
    }

    updateChart(price, result);

    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — simulateCycle(${cycle}) complete`);
}

// --- Start simulation ---
function startSimulation(totalCycles) {
    logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — initBot started for ${totalCycles} cycles (1‑minute interval)`);

    currentCycle = 1;
    cycleTimer = setInterval(() => {
        if (currentCycle > totalCycles) {
            clearInterval(cycleTimer);
            logToPanel(`[INFO] ${new Date().toLocaleTimeString()} — Simulation complete`);
            return;
        }

        simulateCycle(currentCycle);
        currentCycle++;
    }, 60000); // 1 minute
}

// Expose globally
window.startSimulation = startSimulation;
window.simulateCycle = simulateCycle;
window.initBot = startSimulation;
