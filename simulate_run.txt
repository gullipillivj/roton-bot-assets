// simulate_run.txt — orchestrator
logToPanel("simulate_run.txt loaded");

// --- Node spawn disabled in browser context ---

// Replace the spawn call with a simulated coin fetch instead.

let initialBalance = 0; // store starting balance

// --- Helper: simulate coins.txt output ---
async function getCoins() {
    // Simulate the same output coins.txt would produce
    return [{ symbol: "BTCUSDT", address: "BTCUSDT", price: 30215.45 }];
}

// --- Randomize with seed ---
function pickRandomCoin(coins, seed) {
    if (!coins || coins.length === 0) return null;
    const rng = Math.abs(Math.sin(seed)) % 1; // seeded RNG
    const idx = Math.floor(rng * coins.length);
    return coins[idx];
}

// --- PancakeSwap buy (simulated for now) ---
async function pancakeBuy(tokenAddress, amount) {
    logWithTime(`[PancakeSwap] Simulated trade: ${amount} USDT -> ${tokenAddress}`);
    return { status: "success", spent: amount, gas: 0 };
}

// --- Single trading cycle ---
async function simulateCycle(cycleNum) {
    logWithTime(`[Cycle ${cycleNum}] started`);

    const balanceBox = document.getElementById("amountBox");
    let balance = parseFloat(balanceBox.value);

    if (!balance || balance <= 0) {
        logWithTime(`[Cycle ${cycleNum}] Skipped — balance is zero`);
        return;
    }

    const coins = await getCoins();
    const chosenCoin = pickRandomCoin(coins, cycleNum);
    if (!chosenCoin) {
        logWithTime(`[Cycle ${cycleNum}] No coin available`);
        return;
    }

    logWithTime(`[INFO] ${new Date().toLocaleTimeString()} — Cycle ${cycleNum}: Bought ${chosenCoin.symbol} at ${chosenCoin.price} USDT`);

    const result = await pancakeBuy(chosenCoin.address, balance);
    if (result.status === "success") {
        balance -= (result.spent + result.gas);
        balanceBox.value = balance;

        // --- Profit calculation ---
        const profit = balance - initialBalance;
        logWithTime(`[Cycle ${cycleNum}] New balance: ${balance} USDT | Profit: ${profit.toFixed(2)} USDT`);

        updateChart(cycleNum, balance);
    }

    logWithTime(`[Cycle ${cycleNum}] completed`);
}

// --- Start bot ---
function initBot_run(cycles) {
    const balanceBox = document.getElementById("amountBox");
    initialBalance = parseFloat(balanceBox.value) || 0; // capture starting balance

    let cycleNum = 1;
    logWithTime(`[Bot] Starting for ${cycles} cycles with initial balance ${initialBalance} USDT`);

    window.botInterval = setInterval(async () => {
        if (cycleNum > cycles) {
            stopBot_run();
            return;
        }
        await simulateCycle(cycleNum);
        cycleNum++;
    }, 30000); // 30 seconds
}

// --- Stop bot ---
function stopBot_run() {
    if (window.botInterval) {
        clearInterval(window.botInterval);
        logWithTime("[Bot] Stopped");
    }
}

// --- Expose globally for main_run.txt ---
window.initBot_run = initBot_run;
window.stopBot_run = stopBot_run;

// --- Hook up Start/Stop buttons (IDs must match main_run.txt) ---
document.addEventListener("DOMContentLoaded", () => {
    const startBtn = document.getElementById("startBtn");
    const stopBtn = document.getElementById("stopBtn");

    if (startBtn) {
        startBtn.addEventListener("click", () => {
            initBot_run(10); // start with 10 cycles
        });
    }

    if (stopBtn) {
        stopBtn.addEventListener("click", () => {
            stopBot_run();
        });
    }
});
