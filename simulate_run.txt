window.balanceHistory = [];

document.addEventListener("DOMContentLoaded", () => {
    const ctx = document.getElementById("balanceChart").getContext("2d");
    window.balanceChart = new Chart(ctx, {
        type: "bar",
        data: {
            labels: [],
            datasets: [
                {
                    label: "Investment Balance",
                    data: [],
                    backgroundColor: "rgba(243,186,47,0.6)",
                    borderColor: "#f3ba2f",
                    borderWidth: 1
                },
                {
                    label: "Profit/Loss",
                    data: [],
                    backgroundColor: "rgba(0,255,0,0.4)",
                    borderColor: "#00FF00",
                    borderWidth: 1
                }
            ]
        },
        options: {
            responsive: true,
            plugins: { legend: { display: true } },
            scales: {
                x: { title: { display: true, text: "Cycle" } },
                y: { title: { display: true, text: "USDT" } }
            }
        }
    });
});

function updateChart(profit) {
    const cycleLabel = `Cycle ${window.balanceChart.data.labels.length + 1}`;
    window.balanceChart.data.labels.push(cycleLabel);
    window.balanceChart.data.datasets[0].data.push(window.balanceHistory[window.balanceHistory.length - 1]);
    window.balanceChart.data.datasets[1].data.push(profit);
    window.balanceChart.update();
}

if (typeof window.logToPanel !== "function") {
    window.logToPanel = function (msg) { console.log(msg); };
}

function logWithTime(message) {
    const now = new Date();
    const timeStr = now.toLocaleTimeString("en-GB", { hour12: false });
    window.logToPanel(`[INFO] ${timeStr} — ${message}`);
}

// --- Simulation cycle ---
async function simulateCycle(cycleNum) {
    try {
        logWithTime(`[Cycle] ${cycleNum} started`);
        let investBalance = parseFloat(document.getElementById("investBalance").value);
        const reserve = investBalance * 0.1;
        let balance = investBalance - reserve;

        logWithTime(`[Trade] Buying token on PancakeSwap...`);
        await new Promise(r => setTimeout(r, 3000));
        logWithTime(`[Trade] Swap complete for cycle ${cycleNum}`);

        const profit = (Math.random() - 0.4) * 10; // random profit/loss
        investBalance += profit;
        document.getElementById("investBalance").value = investBalance.toFixed(2);
        window.balanceHistory.push(investBalance);
        updateChart(profit);

        logWithTime(`[Cycle] ${cycleNum} complete — Profit ${profit.toFixed(2)} USDT`);
    } catch (err) {
        logWithTime(`[ERROR] simulateCycle(${cycleNum}) crashed: ${err}`);
    }
}

// --- Bot orchestration ---
let runTimer = null;
let currentCycle = 1;

function initBot_run(totalCycles = 10) {
    logWithTime(`[Bot] Starting production run for ${totalCycles} cycles (15‑second interval)`);

    currentCycle = 1;
    simulateCycle(currentCycle);
    currentCycle++;

    runTimer = setInterval(async () => {
        if (currentCycle > totalCycles) {
            clearInterval(runTimer);
            logWithTime("[Bot] Production run complete");
            return;
        }
        await simulateCycle(currentCycle);
        currentCycle++;
    }, 15000); // 15 seconds
}

function stopBot_run() {
    if (runTimer) {
        clearInterval(runTimer);
        logWithTime("[Bot] Run stopped manually");
    }
}

// Expose globally
window.initBot_run = initBot_run;
window.stopBot_run = stopBot_run;
