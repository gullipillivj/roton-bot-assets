async function simulateCycle(userid, cycleNum) {
    try {
        logWithTime(`[Cycle] ${cycleNum} started for ${userid}`);

        let investBalance = parseFloat(document.getElementById("investBalance").value);
        const reserve = investBalance * 0.1;
        let balance = investBalance - reserve;

        logWithTime(`[Trade] Buying token on PancakeSwap...`);
        await new Promise(r => setTimeout(r, 3000));
        logWithTime(`[Trade] Swap complete for cycle ${cycleNum}`);

        // --- Delegate to trade_logic.txt ---
        const profit = await executeTrade(userid, cycleNum);

        // Apply profit/loss
        investBalance += profit;
        document.getElementById("investBalance").value = investBalance.toFixed(2);

        // Update cumulative profit
        window.totalProfit = (window.totalProfit || 0) + profit;
        const profitBox = document.getElementById("liveProfitValue");
        profitBox.textContent = `${window.totalProfit.toFixed(2)} USDT`;
        profitBox.style.color = window.totalProfit >= 0 ? "#00FF00" : "#FF3333";

        window.balanceHistory.push(investBalance);
        updateChart(profit);

        logWithTime(`[Cycle] ${cycleNum} complete — Profit ${profit.toFixed(2)} USDT`);
    } catch (err) {
        logWithTime(`[ERROR] simulateCycle(${cycleNum}) crashed: ${err}`);
    }
}
